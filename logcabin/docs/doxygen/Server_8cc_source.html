45 void
46 Server::handleRPC(RPC::OpaqueServerRPC opaqueRPC)
47 {
48  RPC::ServerRPC rpc(std::move(opaqueRPC));
49  if (!rpc.needsReply()) {
50  // The RPC may have had an invalid header, in which case it needs no
51  // further action.
52  return;
53  }
54  std::shared_ptr<Service> service;
55  {
56  std::unique_lock<std::mutex> lockGuard(mutex);
57  auto it = services.find(rpc.getService());
58  if (it != services.end())
59  service = it->second;
60  }
61  if (service)
62  service->handleRPC(std::move(rpc));
63  else
64  rpc.rejectInvalidService();
65 }
