33 /**
34  * This class represents an asynchronous remote procedure call. Unlike
35  * OpaqueClientRPC, this executes a particular method on a particular service.
36  *
37  * TODO(ongaro): This abstraction should ideally contain protocol version
38  * negotiation for the higher layer. To do so, we'd want to be able to
39  * associate a small amount of state with the session.
40  */

44  /**
45  * Issue an RPC to a remote service.
46  * \param session
47  * A connection to the remote server.
48  * \param service
49  * Identifies the service running on the server.
50  * See Protocol::Common::ServiceId.
51  * \param serviceSpecificErrorVersion
52  * This tells the Service what service-specific errors the client
53  * understands. Clients can expect services to not send
54  * service-specific errors introduced in newer versions, but they
55  * should remain compatible with older versions.
56  * \param opCode
57  * Identifies the remote procedure within the Service to execute.
58  * \param request
59  * The arguments to the remote procedure.
60  */

103  /**
104  * The return type of waitForReply().
105  */
106  enum class Status {
107  /**
108  * The service returned a normal response. This is available in
109  * 'response'.
110  */
111  OK,
112  /**
113  * The service threw an error (but at the transport level, the RPC
114  * succeeded). Service-specific details may be available in
115  * 'serviceSpecificError'.
116  */
117  SERVICE_SPECIFIC_ERROR,
118  /**
119  * The server could not be contacted or did not reply. It is unknown
120  * whether or not the server executed the RPC. More information is
121  * available with getErrorMessage().
122  */
123  RPC_FAILED,
124  };

126  /**
127  * Wait for a reply to the RPC or an error.
128  * Panics if the server responds but is not running the same protocol.
129  *
130  * \param[out] response
131  * If not NULL, this will be filled in if this method returns OK.
132  * \param[out] serviceSpecificError
133  * If not NULL, this will be filled in if this method returns
134  * SERVICE_SPECIFIC_ERROR.
135  * \return
136  * See the individual values of #Status.
137  */
138  Status waitForReply(google::protobuf::Message* response,

141  /**

143  *
144  * All errors indicate that it is unknown whether or not the server
145  * executed the RPC. Unless the RPC was canceled with #cancel(), the

148  *
149  * \return

153  std::string getErrorMessage() const;
