39  void call(OpCode opCode,
40  const google::protobuf::Message& request,
41  google::protobuf::Message& response) {
42  if (opCode == OpCode::OPEN_SESSION) {
43  PC::OpenSession::Response& openSessionResponse =
44  static_cast<PC::OpenSession::Response&>(response);
45  openSessionResponse.set_client_id(1);
46  } else if (opCode == OpCode::READ_ONLY_TREE) {
47  std::unique_lock<std::mutex> lockGuard(mutex);
48  LogCabin::Tree::ProtoBuf::readOnlyTreeRPC(tree,
49  static_cast<const PC::ReadOnlyTree::Request&>(request),
50  static_cast<PC::ReadOnlyTree::Response&>(response));
51  } else if (opCode == OpCode::READ_WRITE_TREE) {
52  std::unique_lock<std::mutex> lockGuard(mutex);
53  LogCabin::Tree::ProtoBuf::readWriteTreeRPC(tree,
54  static_cast<const PC::ReadWriteTree::Request&>(request),
55  static_cast<PC::ReadWriteTree::Response&>(response));
56  } else {
57  PANIC("Unexpected request: %d %s",
58  opCode,
59  Core::ProtoBuf::dumpString(request).c_str());
60  }
61  }
62  std::mutex mutex; ///< prevents concurrent access to 'tree'
63  LogCabin::Tree::Tree tree;
64 };
