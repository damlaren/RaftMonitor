48  if (!to.ParseFromArray(

62 void
63 serialize(const google::protobuf::Message& from,
64  RPC::Buffer& to,
65  uint32_t skipBytes)
66 {
67  // SerializeToArray seems to always return true, so we explicitly check
68  // IsInitialized to make sure all required fields are set.
69  if (!from.IsInitialized()) {
70  PANIC("Missing fields in protocol buffer of type %s: %s",
71  from.GetTypeName().c_str(),
72  from.InitializationErrorString().c_str());
73  }
74  uint32_t length = from.ByteSize();
75  char* data = new char[skipBytes + length];
76  from.SerializeToArray(data + skipBytes, length);
77  to.setData(data, skipBytes + length, RPC::Buffer::deleteArrayFn<char>);
78 }
