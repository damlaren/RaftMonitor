32 /**
33  * This class contains an event loop based on Linux's epoll interface.
34  * It keeps track of interesting events such as timers and socket activity, and
35  * arranges for callbacks to be invoked when the events happen.
36  */

40  /**
41  * Lock objects are used to synchronize between the Event::Loop thread and
42  * other threads. As long as a Lock object exists the following guarantees
43  * are in effect: either
44  * (a) the thread is the event loop thread or
45  * (b) no other thread has a Lock object and the event loop thread has
46  * paused in a safe place (with no event handlers active) waiting for
47  * the Lock to be destroyed.
48  * Locks may be used recursively.
49  */
50  class Lock {
51  public:

54  /// Destructor. Release the lock.
55  ~Lock();

57  Event::Loop& eventLoop;
58  private:
59  // Lock is not copyable.
60  Lock(const Lock&) = delete;
61  Lock& operator=(const Lock&) = delete;
62  };
