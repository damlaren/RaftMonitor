119 
120  std::pair<uint64_t, Configuration> configuration =

127  servers.emplace_back(i + 1, options.servers.at(i));
128  ConfigurationResult result = cluster.setConfiguration(id, servers);
129 
130  std::cout << "Reconfiguration ";
