39 std::string
40 fileToProto(const File& dir, const std::string& path,
41  google::protobuf::Message& out)
42 {
43  FilesystemUtil::File file =
44  FilesystemUtil::tryOpenFile(dir, path, O_RDONLY);
45  if (file.fd == -1) {
46  return format("Could not open %s/%s: %s",
47  dir.path.c_str(), path.c_str(), strerror(errno));
48  }
49  FilesystemUtil::FileContents reader(file);
50 
51  char checksum[Core::Checksum::MAX_LENGTH];
52  uint64_t bytesRead = reader.copyPartial(0, checksum, sizeof(checksum));
53  uint32_t checksumBytes = Core::Checksum::length(checksum,
54  uint32_t(bytesRead));
55  if (checksumBytes == 0)
56  return format("File %s missing checksum", file.path.c_str());
57 
58  uint64_t dataLen = reader.getFileLength() - checksumBytes;
59  const void* data = reader.get(checksumBytes, dataLen);
60  std::string error = Core::Checksum::verify(checksum, data, dataLen);
61  if (!error.empty()) {
62  return format("Checksum verification failure on %s: %s",
63  file.path.c_str(), error.c_str());
64  }
65 
66 #if BINARY_FORMAT
67  RPC::Buffer contents(const_cast<void*>(data), dataLen, NULL);
68  if (!RPC::ProtoBuf::parse(contents, out))
69  return format("Failed to parse protobuf in %s", file.path.c_str());
70 #else
71  std::string contents(static_cast<const char*>(data), dataLen);
72  Core::ProtoBuf::Internal::fromString(contents, out);
73 #endif
74  return "";
75 }

77 FilesystemUtil::File
78 protoToFile(const google::protobuf::Message& in,
79  const File& dir, const std::string& path)
80 {
81  FilesystemUtil::File file =
82  FilesystemUtil::openFile(dir, path, O_CREAT|O_WRONLY|O_TRUNC);
83  const void* data = NULL;
84  uint64_t len = 0;
85 #if BINARY_FORMAT
86  RPC::Buffer contents;
87  RPC::ProtoBuf::serialize(in, contents);
88  data = contents.getData();
89  len = contents.getLenhgt();
90 #else
91  std::string contents(Core::ProtoBuf::dumpString(in));
92  contents = "\n" + contents;
93  data = contents.data();
94  len = uint64_t(contents.length());
95 #endif
96  char checksum[Core::Checksum::MAX_LENGTH];
97  uint32_t checksumLen = Core::Checksum::calculate("SHA-1",
98  data, len,
99  checksum);
100 
101  ssize_t written = FilesystemUtil::write(file.fd, {
102  {checksum, checksumLen},
103  {data, len},
104  });
105  if (written == -1) {
106  PANIC("Failed to write to %s: %s",
107  file.path.c_str(), strerror(errno));
108  }
109 
110  return file;
111 }

146 SimpleFileLog::SimpleFileLog(const FilesystemUtil::File& parentDir)
147  : memoryLog()
148  , metadata()
149  , dir(FilesystemUtil::openDir(parentDir, "log"))
150  , lostAndFound(FilesystemUtil::openDir(dir, "unknown"))
151  , currentSync(new Sync(0))
152 {
153  std::vector<uint64_t> fsEntryIds = getEntryIds();
154 
155  SimpleFileLogMetadata::Metadata metadata1;
156  SimpleFileLogMetadata::Metadata metadata2;
157  std::string error1 = readMetadata("metadata1", metadata1);
158  std::string error2 = readMetadata("metadata2", metadata2);
159  if (error1.empty() && error2.empty()) {
160  if (metadata1.version() > metadata2.version())
161  metadata = metadata1;
162  else
163  metadata = metadata2;
164  } else if (error1.empty()) {
165  metadata = metadata1;
166  } else if (error2.empty()) {
167  metadata = metadata2;
168  } else {
169  // Brand new servers won't have metadata.
170  WARNING("Error reading metadata1: %s", error1.c_str());
171  WARNING("Error reading metadata2: %s", error2.c_str());
172  if (!fsEntryIds.empty()) {
173  PANIC("No readable metadata file but found entries in %s",
174  dir.path.c_str());
175  }
176  metadata.set_entries_start(1);
177  metadata.set_entries_end(0);
178  }
179 
180  std::vector<uint64_t> found;
181  for (auto it = fsEntryIds.begin(); it != fsEntryIds.end(); ++it) {
182  if (*it < metadata.entries_start() || *it > metadata.entries_end())
183  found.push_back(*it);
184  }
185 
186  std::string time;
187  {
188  struct timespec now;
189  clock_gettime(CLOCK_REALTIME, &now);
190  time = format("%010lu.%06lu", now.tv_sec, now.tv_nsec / 1000);
191  }
192  for (auto it = found.begin(); it != found.end(); ++it) {
193  uint64_t entryId = *it;
194  std::string oldName = format("%016lx", entryId);
195  std::string newName = format("%s-%016lx", time.c_str(), entryId);
196  WARNING("Moving extraneous file %s/%s to %s/%s",
197  dir.path.c_str(), oldName.c_str(),
198  lostAndFound.path.c_str(), newName.c_str());
199  FilesystemUtil::rename(dir, oldName,
200  lostAndFound, newName);
201  FilesystemUtil::fsync(lostAndFound);
202  FilesystemUtil::fsync(dir);
203  }
204 
205  memoryLog.truncatePrefix(metadata.entries_start());
206  for (uint64_t id = metadata.entries_start();
207  id <= metadata.entries_end();
208  ++id) {
209  Log::Entry e = read(format("%016lx", id));
210  memoryLog.append({&e});
211  }
212 
213  Log::metadata = metadata.raft_metadata();
214  // Write both metadata files
215  updateMetadata();
216  updateMetadata();
217 }

225 std::pair<uint64_t, uint64_t>
226 SimpleFileLog::append(const std::vector<const Entry*>& entries)
227 {
228  std::pair<uint64_t, uint64_t> range = memoryLog.append(entries);
229  for (uint64_t index = range.first; index <= range.second; ++index) {
230  FilesystemUtil::File file = protoToFile(memoryLog.getEntry(index),
231  dir, format("%016lx", index));
232  currentSync->fds.push_back({file.release(), true});
233  }
234  FilesystemUtil::File mdfile = updateMetadataCallerSync();
235  currentSync->fds.push_back({dir.fd, false});
236  currentSync->fds.push_back({mdfile.release(), true});
237  currentSync->lastIndex = range.second;
238  return range;
239 }

320 std::vector<uint64_t>
321 SimpleFileLog::getEntryIds() const
322 {
323  std::vector<std::string> filenames = FilesystemUtil::ls(dir);
324  std::vector<uint64_t> entryIds;
325  for (auto it = filenames.begin(); it != filenames.end(); ++it) {
326  const std::string& filename = *it;
327  if (filename == "metadata1" ||
328  filename == "metadata2" ||
329  filename == "unknown") {
330  continue;
331  }
332  uint64_t entryId;
333  unsigned bytesConsumed;
334  int matched = sscanf(filename.c_str(), "%016lx%n", // NOLINT
335  &entryId, &bytesConsumed);
336  if (matched != 1 || bytesConsumed != filename.length()) {
337  WARNING("%s doesn't look like a valid entry ID (from %s)",
338  filename.c_str(),
339  (dir.path + "/" + filename).c_str());
340  continue;
341  }
342  entryIds.push_back(entryId);
343  }
344  return entryIds;
345 }
