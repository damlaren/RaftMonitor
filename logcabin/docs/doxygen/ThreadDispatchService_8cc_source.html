45 {
46  // Signal the threads to exit.
47  {

67 ThreadDispatchService::handleRPC(ServerRPC serverRPC)
68 {
69  std::unique_lock<std::mutex> lockGuard(mutex);
70  assert(!exit);
71  rpcQueue.push(std::move(serverRPC));
72  if (numFreeWorkers == 0 && threads.size() < maxThreads)
73  threads.emplace_back(&ThreadDispatchService::workerMain, this);
74  conditionVariable.notify_one();
75 }

83 void
84 ThreadDispatchService::workerMain()
85 {
86  Core::ThreadId::setName(
87  Core::StringUtil::format("%s(%lu)",
88  threadSafeService->getName().c_str(),
89  Core::ThreadId::getId()));
90  while (true) {
91  ServerRPC rpc;
92  { // find an RPC to process
93  std::unique_lock<std::mutex> lockGuard(mutex);
94  ++numFreeWorkers;
95  while (!exit && rpcQueue.empty())
96  conditionVariable.wait(lockGuard);
97  --numFreeWorkers;
98  if (exit)
99  return;
100  rpc = std::move(rpcQueue.front());
101  rpcQueue.pop();
102  }
103  // execute RPC handler
104  threadSafeService->handleRPC(std::move(rpc));
105  }
106 }
