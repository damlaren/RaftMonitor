46 std::string
47 format(const char* format, ...)
48 {
49  std::string s;
50  va_list ap;
51  va_start(ap, format);
52 
53  // We're not really sure how big of a buffer will be necessary.
54  // Try 1K, if not the return value will tell us how much is necessary.
55  int bufSize = 1024;
56  while (true) {
57  char buf[bufSize];
58  // vsnprintf trashes the va_list, so copy it first
59  va_list aq;
60  __va_copy(aq, ap);
61  int r = vsnprintf(buf, bufSize, format, aq);
62  assert(r >= 0); // old glibc versions returned -1
63  if (r < bufSize) {
64  s = buf;
65  break;
66  }
67  bufSize = r + 1;
68  }
69 
70  va_end(ap);
71  return s;
72 }

96  while (true) {
97  size_t replacePos = haystack.find(needle, startPos);
98  if (replacePos == haystack.npos)
99  return;
100  haystack.replace(replacePos, needle.length(), replacement);
101  startPos = replacePos + replacement.length();
