32 /**
33  * This interface is used by RaftConsensus to store log entries and metadata.
34  * Typically, implementations will persist the log entries and metadata to
35  * stable storage (but MemoryLog keeps it all in volatile memory).
36  */
37 class Log {
38  public:
39  /**
40  * An interface for flushing newly appended log entries to stable storage.
41  * Leaders do this in a separate thread, while followers and candidates do
42  * this immediately after appending the entries.
43  *
44  * Callers should wait() on all Sync objects prior to calling
45  * truncateSuffix(). This never happens on leaders, so it's not a real
46  * limitation, but things may go wonky otherwise.
47  */
48  class Sync {
49  public:
50  explicit Sync(uint64_t lastIndex);
51  virtual ~Sync();
52  /**
53  * Wait for the log entries to be durable.
54  * This is safe to call while the Log is being accessed and modified
55  * from a separate thread.
56  * PANICs on errors.
57  */
58  virtual void wait() {}
59  /**
60  * The index of the last log entry that is being flushed.
61  * After the call to wait, every entry in the log up to this one is
62  * durable.
63  */
64  uint64_t lastIndex;
65  /**
66  * Used by destructor to make sure that Log::syncComplete was called.
67  */
68  bool completed;
69  };
