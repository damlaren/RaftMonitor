58 void
59 Timer::schedule(uint64_t nanoseconds)
60 {
61  // avoid accidental de-schedules: epoll's semantics are that a timer for 0
62  // seconds and 0 nanoseconds will never fire.
63  if (nanoseconds == 0)
64  nanoseconds = 1;
65 
66  const uint64_t nanosPerSecond = 1000 * 1000 * 1000;
67  struct itimerspec newValue;
68  memset(&newValue, 0, sizeof(newValue));
69  newValue.it_value.tv_sec = nanoseconds / nanosPerSecond;
70  newValue.it_value.tv_nsec = nanoseconds % nanosPerSecond;
71  int r = timerfd_settime(fd, 0, &newValue, NULL);
72  if (r != 0) {
73  PANIC("Could not set timer to +%luns: %s",
74  nanoseconds,
75  strerror(errno));
76  }
77 }
