30 /**
31  * A Server listens for incoming RPCs over TCP connections and dispatches these
32  * to Services.
33  * Servers can be created from any thread, but they will always run on
34  * the thread running the Event::Loop. Services will always run on a thread
35  * pool.
36  */
37 class Server : public OpaqueServer {
38  public:
39  /**
40  * Constructor. This object won't actually do anything until bind() is
41  * called.
42  * \param eventLoop
43  * Event::Loop that will be used to find out when the underlying
44  * socket may be read from or written to without blocking.
45  * \param maxMessageLength
46  * The maximum number of bytes to allow per request/response. This
47  * exists to limit the amount of buffer space a single RPC can use.
48  * Attempting to send longer responses will PANIC; attempting to
49  * receive longer requests will disconnect the underlying socket.
50  */
51  Server(Event::Loop& eventLoop, uint32_t maxMessageLength);
52 
53  /**
54  * Destructor. ServerRPC objects originating from this Server may be kept
55  * around after this destructor returns; however, they won't actually send
56  * replies anymore.
57  */
58  ~Server();
59 
60  /**
61  * Register a Service to receive RPCs from clients. If a service has
62  * already been registered for this service ID, this will replace it. This
63  * may be called from any thread.
64  * \param serviceId
65  * A unique ID for the service. See Protocol::Common::ServiceId.
66  * \param service
67  * The service to invoke when RPCs arrive with the given serviceId.
68  * This service will always be invoked on a thread pool.
69  * \param maxThreads
70  * The maximum number of threads to execute RPCs concurrently inside
71  * the service.
72  */
73  void registerService(uint16_t serviceId,
74  std::shared_ptr<Service> service,
75  uint32_t maxThreads);

77  private:
78  /**
79  * This is called by the base class, OpaqueServer, when an RPC arrives.
80  */
81  void handleRPC(RPC::OpaqueServerRPC opaqueRPC);
