39  * An OpaqueServer listens for incoming RPCs over TCP connections.
40  * OpaqueServers can be created from any thread, but they will always run on
41  * the thread running the Event::Loop.
42  */
43 class OpaqueServer {
44  public:
45  /**
46  * Constructor. This object won't actually do anything until bind() is
47  * called.
48  * \param eventLoop
49  * Event::Loop that will be used to find out when the underlying
50  * socket may be read from or written to without blocking.
51  * \param maxMessageLength
52  * The maximum number of bytes to allow per request/response. This
53  * exists to limit the amount of buffer space a single RPC can use.
54  * Attempting to send longer responses will PANIC; attempting to
55  * receive longer requests will disconnect the underlying socket.
56  */
57  OpaqueServer(Event::Loop& eventLoop, uint32_t maxMessageLength);

60  * Destructor. OpaqueServerRPC objects originating from this OpaqueServer
61  * may be kept around after this destructor returns; however, they won't
62  * actually send replies anymore.
63  */
64  virtual ~OpaqueServer();

66  /**
67  * Listen on an address for new client connections. You can call this
68  * multiple times to listen on multiple addresses. (But if you call this
69  * twice with the same address, the second time will always throw an
70  * error.)
71  * This method is thread-safe.
72  * \param listenAddress
73  * The TCP address on listen for new client connections.
74  * \return
75  * An error message if this was not able to listen on the given
76  * address; the empty string otherwise.
77  */
