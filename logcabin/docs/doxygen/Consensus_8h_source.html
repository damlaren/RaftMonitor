75  enum {

77  * This is a normal entry containing a client request for the state
78  * machine. The 'data' field contains that request, and
79  * 'snapshotReader' is not set.
80  */
81  DATA,
82  /**
83  * This is a snapshot: the state machine should clear its state and

85  * 'snapshotReader' should be used to read the snapshot contents
86  * from.
87  */
88  SNAPSHOT,
89  /**
90  * Some entries should be ignored by the state machine (they are
91  * consumed internally by the consensus module). For client service
92  * threads to know when a state machine is up-to-date, it's easiest
93  * for the state machine to get empty entries back for these, and
94  * simply call back into getNextEntry() again with the next ID,
95  * Entries of type 'SKIP' will have neither 'data' nor
96  * 'snapshotReader' set.
97  */

128  /**
