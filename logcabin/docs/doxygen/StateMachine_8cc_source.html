83 bool
84 StateMachine::getResponse(const PC::ExactlyOnceRPCInfo& rpcInfo,
85  PC::CommandResponse& response) const
86 {
87  std::unique_lock<std::mutex> lockGuard(mutex);
88  auto sessionIt = sessions.find(rpcInfo.client_id());
89  if (sessionIt == sessions.end()) {
90  WARNING("Client %lu session expired but client still active",
91  rpcInfo.client_id());
92  return false;
93  }
94  const Session& session = sessionIt->second;
95  auto responseIt = session.responses.find(rpcInfo.rpc_number());
96  if (responseIt == session.responses.end()) {
97  // The response for this RPC has already been removed: the client is
98  // not waiting for it. This request is just a duplicate that is safe to
99  // drop.
100  WARNING("Client %lu asking for discarded response to RPC %lu",
101  rpcInfo.client_id(), rpcInfo.rpc_number());
102  return false;
103  }
104  response = responseIt->second;
105  return true;
106 }

127 void
128 StateMachine::apply(uint64_t entryId, const std::string& data)
129 {
130  // TODO(ongaro): Switch from string to binary format. This is probably
131  // really slow to parse.
132  PC::Command command = Core::ProtoBuf::fromString<PC::Command>(data);
133  Session* session = NULL;
134  if (command.has_tree()) {
135  PC::ExactlyOnceRPCInfo rpcInfo = command.tree().exactly_once();
136  auto it = sessions.find(rpcInfo.client_id());
137  if (it == sessions.end()) {
138  // session does not exist
139  } else {
140  // session exists
141  session = &it->second;
142  expireResponses(*session, rpcInfo.first_outstanding_rpc());
143  if (rpcInfo.rpc_number() < session->firstOutstandingRPC) {
144  // response already discarded, do not re-apply
145  } else {
146  auto inserted = session->responses.insert(
147  {rpcInfo.rpc_number(), {}});
148  if (inserted.second) {
149  // response not found, apply and save it
150  Tree::ProtoBuf::readWriteTreeRPC(
151  tree,
152  command.tree(),
153  *inserted.first->second.mutable_tree());
154  } else {
155  // response exists, do not re-apply
156  }
157  }
158  }
159  } else if (command.has_open_session()) {
160  uint64_t clientId = entryId;
161  session = &sessions.insert({clientId, {}}).first->second;
162  } else {
163  PANIC("unknown command at %lu: %s", entryId, data.c_str());
164  }
165 
166  if (command.has_nanoseconds_since_epoch()) {
167  if (session != NULL) {
168  session->lastModified = command.nanoseconds_since_epoch();
169  session = NULL; // pointer invalidated by expireSessions()
170  }
171  expireSessions(command.nanoseconds_since_epoch());
172  }
173 }

175 void
176 StateMachine::applyThreadMain()
177 {
178  Core::ThreadId::setName("StateMachine");
179  try {
180  while (true) {
181  Consensus::Entry entry = consensus->getNextEntry(lastEntryId);
182  std::unique_lock<std::mutex> lockGuard(mutex);
183  switch (entry.type) {
184  case Consensus::Entry::SKIP:
185  break;
186  case Consensus::Entry::DATA:
187  apply(entry.entryId, entry.data);
188  break;
189  case Consensus::Entry::SNAPSHOT:
190  NOTICE("Loading snapshot through entry %lu into "
191  "state machine", entry.entryId);
192  loadSessionSnapshot(entry.snapshotReader->getStream());
193  tree.loadSnapshot(entry.snapshotReader->getStream());
194  break;
195  }
196  lastEntryId = entry.entryId;
197  entriesApplied.notify_all();
198  if (shouldTakeSnapshot(lastEntryId))
199  snapshotSuggested.notify_all();
200  }
201  } catch (const ThreadInterruptedException& e) {
202  NOTICE("exiting");
203  std::unique_lock<std::mutex> lockGuard(mutex);
204  exiting = true;
205  entriesApplied.notify_all();
206  snapshotSuggested.notify_all();
207  if (childPid != 0) {
208  int r = kill(childPid, SIGHUP);
209  if (r != 0) {
210  WARNING("Could not send SIGHUP to child process: %s",
211  strerror(errno));
212  }
213  }
214  }

217 void
218 StateMachine::dumpSessionSnapshot(
219  google::protobuf::io::CodedOutputStream& stream) const
220 {
221  // dump into protobuf
222  SessionsProto::Sessions sessionsProto;
223  for (auto it = sessions.begin(); it != sessions.end(); ++it) {
224  SessionsProto::Session& session = *sessionsProto.add_session();
225  session.set_client_id(it->first);
226  session.set_last_modified(it->second.lastModified);
227  session.set_first_outstanding_rpc(it->second.firstOutstandingRPC);
228  for (auto it2 = it->second.responses.begin();
229  it2 != it->second.responses.end();
230  ++it2) {
231  SessionsProto::Response& response = *session.add_rpc_response();
232  response.set_rpc_number(it2->first);
233  *response.mutable_response() = it2->second;
234  }
235  }
236 
237  // write protobuf to stream
238  int size = sessionsProto.ByteSize();
239  stream.WriteLittleEndian32(size);
240  sessionsProto.SerializeWithCachedSizes(&stream);
241 }

243 void
244 StateMachine::expireResponses(Session& session, uint64_t firstOutstandingRPC)
245 {
246  if (session.firstOutstandingRPC >= firstOutstandingRPC)
247  return;
248  session.firstOutstandingRPC = firstOutstandingRPC;
249  auto it = session.responses.begin();
250  while (it != session.responses.end()) {
251  if (it->first < session.firstOutstandingRPC)
252  it = session.responses.erase(it);
253  else
254  ++it;
255  }

258 void
259 StateMachine::expireSessions(uint64_t nanosecondsSinceEpoch)
260 {
261  auto it = sessions.begin();
262  while (it != sessions.end()) {
263  Session& session = it->second;
264  uint64_t expireTime = session.lastModified + sessionTimeoutNanos;
265  if (expireTime < nanosecondsSinceEpoch) {
266  uint64_t diffNanos = nanosecondsSinceEpoch - session.lastModified;
267  NOTICE("Expiring client %lu's session after %lu.%09lu seconds "
268  "due to inactivity",
269  it->first,
270  diffNanos / (1000 * 1000 * 1000UL),
271  diffNanos % (1000 * 1000 * 1000UL));
272  it = sessions.erase(it);
273  } else {
274  ++it;
275  }
276  }

280 void
281 StateMachine::loadSessionSnapshot(
282  google::protobuf::io::CodedInputStream& stream)
283 {
284  // read protobuf from stream
285  bool ok = true;
286  uint32_t numBytes = 0;
287  ok = stream.ReadLittleEndian32(&numBytes);
288  if (!ok)
289  PANIC("couldn't read snapshot");
290  SessionsProto::Sessions sessionsProto;
291  auto limit = stream.PushLimit(numBytes);
292  ok = sessionsProto.MergePartialFromCodedStream(&stream);
293  stream.PopLimit(limit);
294  if (!ok)
295  PANIC("couldn't read snapshot");
296 
297  // load from protobuf
298  sessions.clear();
299  for (auto it = sessionsProto.session().begin();
300  it != sessionsProto.session().end();
301  ++it) {
302  Session& session = sessions.insert({it->client_id(), {}})
303  .first->second;
304  session.lastModified = it->last_modified();
305  session.firstOutstandingRPC = it->first_outstanding_rpc();
306  for (auto it2 = it->rpc_response().begin();
307  it2 != it->rpc_response().end();
308  ++it2) {
309  session.responses.insert({it2->rpc_number(), it2->response()});
310  }
311  }

314 bool
315 StateMachine::shouldTakeSnapshot(uint64_t lastIncludedIndex) const
316 {
317  SnapshotStats::SnapshotStats stats = consensus->getSnapshotStats();
318  if (stats.log_bytes() < snapshotMinLogSize)
319  return false;
320  if (stats.log_bytes() < stats.last_snapshot_bytes() * snapshotRatio)
321  return false;
322  if (lastIncludedIndex < stats.last_snapshot_index())
323  return false;
324  if (lastIncludedIndex < stats.last_log_index() * 3 / 4)
325  return false;
326  return true;
327 }

344 void
345 StateMachine::takeSnapshot(uint64_t lastIncludedIndex,
346  std::unique_lock<std::mutex>& lockGuard)
347 {
348  // Open a snapshot file, then fork a child to write a consistent view of
349  // the state machine to the snapshot file while this process continues
350  // accepting requests.
351  std::unique_ptr<Storage::SnapshotFile::Writer> writer =
352  consensus->beginSnapshot(lastIncludedIndex);
353  // Flush the outstanding changes to the snapshot now so that they
354  // aren't somehow double-flushed later.
355  writer->flushToOS();
356  pid_t pid = fork();
357  if (pid == -1) { // error
358  PANIC("Couldn't fork: %s", strerror(errno));
359  } else if (pid == 0) { // child
360  usleep(stateMachineChildSleepMs * 1000); // for testing purposes
361  dumpSessionSnapshot(writer->getStream());
362  tree.dumpSnapshot(writer->getStream());
363  // Flush the changes to the snapshot file before exiting.
364  writer->flushToOS();
365  _exit(0);
366  } else { // parent
367  assert(childPid == 0);
368  childPid = pid;
369  int status = 0;
370  {
371  // release the lock while blocking on the child to allow
372  // parallelism
373  Core::MutexUnlock<std::mutex> unlockGuard(lockGuard);
374  pid = waitpid(pid, &status, 0);
375  }
376  childPid = 0;
377  if (pid == -1)
378  PANIC("Couldn't waitpid: %s", strerror(errno));
379  if (WIFEXITED(status) && WEXITSTATUS(status) == 0) {
380  NOTICE("Child completed writing state machine contents to "
381  "snapshot staging file");
382  consensus->snapshotDone(lastIncludedIndex, std::move(writer));
383  } else if (exiting &&
384  WIFSIGNALED(status) && WTERMSIG(status) == SIGHUP) {
385  writer->discard();
386  NOTICE("Child exited from SIGHUP since this process is "
387  "exiting");
388  } else {
389  writer->discard();
390  PANIC("Snapshot creation failed with status %d", status);
391  }
392  }
393 }
