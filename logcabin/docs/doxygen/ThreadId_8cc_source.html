65 /**
66  * Return a unique identifier associated with this thread. The
67  * return value has two properties:
68  * - It will never be zero.

70  * been returned this value or ever will be returned this value)

80 void
81 setName(const std::string& name)
82 {
83  // get the thread ID before locking to avoid deadlock
84  uint64_t id = getId();
85  std::unique_lock<std::mutex> lockGuard(Internal::mutex);
86  if (name.empty())
87  Internal::threadNames.erase(id);
88  else
89  Internal::threadNames[id] = name;
90  // set system thread name, useful for gdb
91  // name is truncated at first 16 characters
92  prctl(PR_SET_NAME, name.c_str(), 0, 0, 0);
93 }

95 std::string
96 getName()
97 {
98  // get the thread ID before locking to avoid deadlock
99  uint64_t id = getId();
100  std::unique_lock<std::mutex> lockGuard(Internal::mutex);
101  auto it = Internal::threadNames.find(id);
102  if (it == Internal::threadNames.end())
103  return StringUtil::format("thread %lu", id);
104  else
105  return it->second;
106 }
