111 void
112 OpaqueClientRPC::waitForReply()
113 {
114  std::unique_lock<std::mutex> mutexGuard(mutex);
115  if (ready)
116  return;
117  if (session) {
118  {
119  // release the mutex while calling wait()
120  Core::MutexUnlock<std::mutex> unlockGuard(mutexGuard);
121  session->wait(*this);
122  }
123  update();
124  } else {
125  ready = true;
126  errorMessage = "This RPC was never associated with a ClientSession.";
127  }
128 }
