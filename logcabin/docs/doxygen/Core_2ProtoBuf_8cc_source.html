86 std::string
87 dumpString(const google::protobuf::Message& protoBuf,
88  bool forCopyingIntoTest)
89 {
90  std::string output;
91  google::protobuf::TextFormat::Printer printer;
92  if (forCopyingIntoTest) {
93  // Most lines that use these strings will look like this:
94  // ^ EXPECT_EQ(...,$
95  // ^ "..."
96  // ^ "...");
97  // 12345678901234
98  // Therefore, we want 14 leading spaces. Tell gtest we want 16, though,
99  // so that when we add in the surrounding quotes later, lines won't
100  // wrap.
101  printer.SetInitialIndentLevel(8);
102  }
103  printer.SetUseShortRepeatedPrimitives(true);
104  printer.PrintToString(protoBuf, &output);
105  if (forCopyingIntoTest) {
106  // TextFormat::Printer escapes ' already.
107  replaceAll(output, "\"", "'");
108  replaceAll(output, " ", " \"");
109  replaceAll(output, "\n", "\"\n");
110  }
111  if (!protoBuf.IsInitialized()) {
112  std::vector<std::string> errors;
113  protoBuf.FindInitializationErrors(&errors);
114  std::ostringstream outputBuf;
115  outputBuf << output;
116  for (auto it = errors.begin(); it != errors.end(); ++it) {
117  if (forCopyingIntoTest) {
118  outputBuf << " \"" << *it << ": UNDEFINED\"\n";
119  } else {
120  outputBuf << *it << ": UNDEFINED\n";
121  }
122  }
123  return output + outputBuf.str();
124  }
125  return output;
126 }
