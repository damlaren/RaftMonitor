228 
229  /**
230  * Used to generate error messages during path lookup.
231  * \param end
232  * The last component of 'parents' to include in the returned string;
233  * this is typically the component that caused an error in path
234  * traversal.
235  * \return
236  * The prefix of 'parents' up to and including the given end position.
237  * This is returned as a slash-delimited string not including "/root".
238  */
239  std::string
240  parentsThrough(std::vector<std::string>::const_iterator end) const;
241 
242  public:
243  /**
244  * Status and error message from the constructor. Possible errors are:
245  * - INVALID_ARGUMENT if path is malformed.
246  */
247  Result result;
248 
249  /**
250  * The exact argument given to the constructor.
251  */
252  std::string symbolic;
253  /**
254  * The directories needed to traverse to get to the target.
255  * This usually begins with "root" to get from the super root to the root
256  * directory, then includes the components of the symbolic path up to but
257  * not including the target. If the symbolic path is "/", this will be
258  * empty.
259  */
260  std::vector<std::string> parents;
261  /**
262  * The final component of the path.
263  * This is usually at the end of the symbolic path. If the symbolic path is
264  * "/", this will be "root", used to get from the super root to the root
265  * directory.
266  */
267  std::string target;
