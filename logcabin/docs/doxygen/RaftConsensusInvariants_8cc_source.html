93 void
94 Invariants::checkBasic()
95 {
96  // Log terms monotonically increase
97  uint64_t lastTerm = 0;
98  for (uint64_t entryId = consensus.log->getLogStartIndex();
99  entryId <= consensus.log->getLastLogIndex();
100  ++entryId) {
101  const Storage::Log::Entry& entry = consensus.log->getEntry(entryId);
102  expect(entry.term() >= lastTerm);
103  lastTerm = entry.term();
104  }
105  // The terms in the log do not exceed currentTerm
106  expect(lastTerm <= consensus.currentTerm);
107 
108  // The current configuration should be the last one found in the log
109  bool found = false;
110  for (uint64_t entryId = consensus.log->getLastLogIndex();
111  entryId >= consensus.log->getLogStartIndex();
112  --entryId) {
113  const Storage::Log::Entry& entry = consensus.log->getEntry(entryId);
114  if (entry.type() == Protocol::Raft::EntryType::CONFIGURATION) {
115  expect(consensus.configuration->id == entryId);
116  expect(consensus.configuration->state !=
117  Configuration::State::BLANK);
118  found = true;
119  break;
120  }
121  }
122  if (!found) {
123  if (consensus.log->getLogStartIndex() == 1) {
124  expect(consensus.configuration->id == 0);
125  expect(consensus.configuration->state ==
126  Configuration::State::BLANK);
127  } else {
128  expect(consensus.configuration->id <= consensus.lastSnapshotIndex);
129  }
130  }
131 
132  // Every configuration present in the log should also be present in the
133  // configurationDescriptions map.
134  for (uint64_t entryId = consensus.log->getLogStartIndex();
135  entryId <= consensus.log->getLastLogIndex();
136  ++entryId) {
137  const Storage::Log::Entry& entry = consensus.log->getEntry(entryId);
138  if (entry.type() == Protocol::Raft::EntryType::CONFIGURATION) {
139  auto it = consensus.configurationManager->
140  descriptions.find(entryId);
141  expect(it != consensus.configurationManager->descriptions.end());
142  if (it != consensus.configurationManager->descriptions.end())
143  expect(it->second == entry.configuration());
144  }
145  }
146  // The configuration descriptions map shouldn't have anything past the
147  // snapshot and the log.
148  expect(consensus.configurationManager->descriptions.upper_bound(
149  std::max(consensus.log->getLastLogIndex(),
150  consensus.lastSnapshotIndex)) ==
151  consensus.configurationManager->descriptions.end());
152 
153  // Servers with blank configurations should remain passive. Since the first
154  // entry in every log is a configuration, they should also have empty logs.
155  if (consensus.configuration->state == Configuration::State::BLANK) {
156  expect(consensus.state == RaftConsensus::State::FOLLOWER);
157  expect(consensus.log->getLastLogIndex() == 0);
158  }
159 
160  // The last snapshot covers a committed range.
161  expect(consensus.commitIndex >= consensus.lastSnapshotIndex);
162 
163  // The commitIndex doesn't exceed the length of the log/snapshot.
164  expect(consensus.commitIndex <= consensus.log->getLastLogIndex());
165 
166  // The last log index points at least through the end of the last snapshot.
167  expect(consensus.log->getLastLogIndex() >= consensus.lastSnapshotIndex);
168 
169  // lastLogIndex is either just below the log start (for empty logs) or
170  // larger (for non-empty logs)
171  assert(consensus.log->getLastLogIndex() >=
172  consensus.log->getLogStartIndex() - 1);
173 
174  // advanceCommittedId is called everywhere it needs to be.
175  if (consensus.state == RaftConsensus::State::LEADER) {
176  uint64_t majorityEntry =
177  consensus.configuration->quorumMin(&Server::getLastAgreeIndex);
178  expect(consensus.commitIndex >= majorityEntry ||
179  majorityEntry < consensus.log->getLogStartIndex() ||
180  consensus.log->getEntry(majorityEntry).term() !=
181  consensus.currentTerm);
182  }
183 
184  // A leader always points its leaderId at itself.
185  if (consensus.state == RaftConsensus::State::LEADER)
186  expect(consensus.leaderId == consensus.serverId);
187 
188  // A leader always voted for itself. (Candidates can vote for others when
189  // they abort an election.)
190  if (consensus.state == RaftConsensus::State::LEADER) {
191  expect(consensus.votedFor == consensus.serverId);
192  }
193 
194  // A follower and candidate always has a timer set; a leader has it at
195  // TimePoint::max().
196  if (consensus.state == RaftConsensus::State::LEADER) {
197  expect(consensus.startElectionAt == TimePoint::max());
198  } else {
199  expect(consensus.startElectionAt > TimePoint::min());
200  expect(consensus.startElectionAt <=
201  Clock::now() + std::chrono::milliseconds(
202  RaftConsensus::ELECTION_TIMEOUT_MS * 2));
203  }
204 
205  // Log metadata is updated when the term or vote changes.
206  expect(consensus.log->metadata.current_term() == consensus.currentTerm);
207  expect(consensus.log->metadata.voted_for() == consensus.votedFor);
208 }

210 void
211 Invariants::checkDelta()
212 {
213  if (!previous) {
214  previous.reset(new ConsensusSnapshot(consensus));
215  return;
216  }
217  std::unique_ptr<ConsensusSnapshot> current(
218  new ConsensusSnapshot(consensus));
219  // Within a term, ...
220  if (previous->currentTerm == current->currentTerm) {
221  // the leader is set at most once.
222  if (previous->leaderId != 0)
223  expect(previous->leaderId == current->leaderId);
224  // the vote is set at most once.
225  if (previous->votedFor != 0)
226  expect(previous->votedFor == current->votedFor);
227  // a leader stays a leader.
228  if (previous->state == RaftConsensus::State::LEADER)
229  expect(current->state == RaftConsensus::State::LEADER);
230  }
231 
232  // Once exiting is set, it doesn't get unset.
233  if (previous->exiting)
234  expect(current->exiting);
235 
236  // These variables monotonically increase.
237  expect(previous->currentTerm <= current->currentTerm);
238  expect(previous->commitIndex <= current->commitIndex);
239  expect(previous->currentEpoch <= current->currentEpoch);
240 
241  // Change requires condition variable notification:
242  if (previous->stateChangedCount == current->stateChangedCount) {
243  expect(previous->currentTerm == current->currentTerm);
244  expect(previous->state == current->state);
245  expect(previous->lastLogIndex == current->lastLogIndex);
246  expect(previous->lastLogTerm == current->lastLogTerm);
247  expect(previous->commitIndex == current->commitIndex);
248  expect(previous->exiting == current->exiting);
249  expect(previous->numPeerThreads <= current->numPeerThreads);
250  expect(previous->configurationId == current->configurationId);
251  expect(previous->configurationState == current->configurationState);
252  expect(previous->startElectionAt == current->startElectionAt);
253  // TODO(ongaro):
254  // an acknowledgement from a peer is received.
255  // a server goes from not caught up to caught up.
256  }
257 
258  previous = std::move(current);
259 }

261 void
262 Invariants::checkPeerBasic()
263 {
264  for (auto it = consensus.configuration->knownServers.begin();
265  it != consensus.configuration->knownServers.end();
266  ++it) {
267  Peer* peer = dynamic_cast<Peer*>(it->second.get()); // NOLINT
268  if (peer == NULL)
269  continue;
270  if (consensus.exiting)
271  expect(peer->exiting);
272  if (!peer->requestVoteDone) {
273  expect(!peer->haveVote_);
274  }
275  expect(peer->lastAgreeIndex <= consensus.log->getLastLogIndex());
276  expect(peer->lastAckEpoch <= consensus.currentEpoch);
277  expect(peer->nextHeartbeatTime <= Clock::now() +
278  std::chrono::milliseconds(consensus.HEARTBEAT_PERIOD_MS));
279  expect(peer->backoffUntil <= Clock::now() +
280  std::chrono::milliseconds(consensus.RPC_FAILURE_BACKOFF_MS));
281 
282  // TODO(ongaro): anything about catchup?
283  }
