97 void
98 Writer::flushToOS()
99 {
100  if (file.fd < 0)
101  PANIC("File already closed");
102  // 1. Destroy the old CodedOutputStream.
103  codedStream.reset();
104  // 2. Flush the FileOutputStream.
105  fileStream->Flush();

111 uint64_t
112 Writer::save()
113 {
114  if (file.fd < 0)
115  PANIC("File already closed");
116  codedStream.reset();
117  fileStream->Flush();
118  fileStream.reset();
119  FilesystemUtil::fsync(file);
120  uint64_t fileSize = FilesystemUtil::getSize(file);
121  file.close();
122  FilesystemUtil::rename(parentDir, stagingName,
123  parentDir, "snapshot");
124  FilesystemUtil::fsync(parentDir);
125  return fileSize;
