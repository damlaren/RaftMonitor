46 void
47 LeaderRPC::call(OpCode opCode,
48  const google::protobuf::Message& request,
49  google::protobuf::Message& response)
50 {
51  typedef RPC::ClientRPC::Status Status;
52 
53  // TODO(ongaro): Rate limit the retries so as not to overwhelm servers
54  // while they're choosing a new leader, etc.
55  while (true) {
56  // Save a reference to the leaderSession
57  std::shared_ptr<RPC::ClientSession> cachedSession;
58  {
59  std::unique_lock<std::mutex> lockGuard(mutex);
60  cachedSession = leaderSession;
61  }
62 
63  // Execute the RPC
64  RPC::ClientRPC rpc(cachedSession,
65  Protocol::Common::ServiceId::CLIENT_SERVICE,
66  1,
67  opCode,
68  request);
69  Protocol::Client::Error serviceSpecificError;
70  Status status = rpc.waitForReply(&response, &serviceSpecificError);
71 
72  // Decode the response
73  switch (status) {
74  case Status::OK:
75  return;
76  case Status::SERVICE_SPECIFIC_ERROR:
77  handleServiceSpecificError(cachedSession,
78  serviceSpecificError);
79  break;
80  case Status::RPC_FAILED:
81  // If the session is broken, get a new one and try again.
82  connectRandom(cachedSession);
83  break;
84  }
85  }
86 }

88 void
89 LeaderRPC::handleServiceSpecificError(
90  std::shared_ptr<RPC::ClientSession> cachedSession,
91  const Protocol::Client::Error& error)
92 {
93  switch (error.error_code()) {
94  case Protocol::Client::Error::NOT_LEADER:
95  // The server we tried is not the current cluster leader.
96  if (error.has_leader_hint()) {
97  // Server returned hint as to who the leader might be.
98  VERBOSE("Trying suggested %s as new leader (was using %s)",
99  error.leader_hint().c_str(),
100  cachedSession->toString().c_str());
101  connectHost(error.leader_hint(), cachedSession);
102  } else {
103  // Well, this server isn't the leader. Try someone else.
104  VERBOSE("Trying random host as new leader (was using %s)",
105  cachedSession->toString().c_str());
106  connectRandom(cachedSession);
107  }
108  break;
109  case Protocol::Client::Error::SESSION_EXPIRED:
110  PANIC("Session expired");
111  default:
112  // Hmm, we don't know what this server is trying to tell us, but
113  // something is wrong. The server shouldn't reply back with error
114  // codes we don't understand. That's why we gave it a
115  // serverSpecificErrorVersion number in the request header.
116  PANIC("Unknown error code %u returned in service-specific error. "
117  "This probably indicates a bug in the server.",
118  error.error_code());
119  }
120 }

133 LeaderRPC::connectRandom(std::shared_ptr<RPC::ClientSession> cachedSession)
134 {
135  std::unique_lock<std::mutex> lockGuard(mutex);
136  if (cachedSession == leaderSession) {
137  // Hope the next random host is the leader.
138  // If that turns out to be false, we will soon find out.
139  hosts.refresh();
140  connect(hosts, lockGuard);
