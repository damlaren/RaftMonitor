24 /**
25  * The number of milliseconds to wait until the client gets suspicious about
26  * the server not responding. After this amount of time elapses, the client
27  * will send a ping to the server. If no response is received within another
28  * TIMEOUT_MS milliseconds, the session is closed.
29  *
30  * TODO(ongaro): How should this value be chosen?
31  * Ideally, you probably want this to be set to something like the 99-th
32  * percentile of your RPC latency.
33  *
34  * TODO(ongaro): How does this interact with TCP?
35  */
36 enum { TIMEOUT_MS = 100 };

58 void
59 ClientSession::ClientMessageSocket::onReceiveMessage(MessageId messageId,
60  Buffer message)
61 {
62  std::unique_lock<std::mutex> mutexGuard(session.mutex);
63 
64  if (messageId == PING_MESSAGE_ID) {
65  if (session.numActiveRPCs > 0 && session.activePing) {
66  // The server has shown that it is alive for now.
67  // Let's get suspicious again in another TIMEOUT_MS.
68  session.activePing = false;
69  session.timer.schedule(TIMEOUT_MS * 1000 * 1000);
70  } else {
71  VERBOSE("Received an unexpected ping response. This can happen "
72  "for a number of reasons and is no cause for alarm. For "
73  "example, this happens if a ping request was sent out, "
74  "then all RPCs completed before the ping response "
75  "arrived.");
76  }
77  return;
78  }
79 
80  auto it = session.responses.find(messageId);
81  if (it == session.responses.end()) {
82  VERBOSE("Received an unexpected response with message ID %lu. "
83  "This can happen for a number of reasons and is no cause "
84  "for alarm. For example, this happens if the RPC was "
85  "cancelled before its response arrived.",
86  messageId);
87  return;
88  }
89  Response& response = *it->second;
90  if (response.status == Response::HAS_REPLY) {
91  WARNING("Received a second response from the server for "
92  "message ID %lu. This indicates that either the client or "
93  "server is assigning message IDs incorrectly, or "
94  "the server is misbehaving. Dropped this response.",
95  messageId);
96  return;
97  }
98 
99  // Book-keeping for timeouts
100  --session.numActiveRPCs;
101  if (session.numActiveRPCs == 0)
102  session.timer.deschedule();
103  else
104  session.timer.schedule(TIMEOUT_MS * 1000 * 1000);
105 
106  // Fill in the response
107  response.status = Response::HAS_REPLY;
108  response.reply = std::move(message);
109  response.ready.notify_all();
110 }

112 void
113 ClientSession::ClientMessageSocket::onDisconnect()
114 {
115  VERBOSE("Disconnected from server %s",
116  session.address.toString().c_str());
117  std::unique_lock<std::mutex> mutexGuard(session.mutex);
118  if (session.errorMessage.empty()) {
119  // Fail all current and future RPCs.
120  session.errorMessage = ("Disconnected from server " +
121  session.address.toString());
122  // Notify any waiting RPCs.
123  for (auto it = session.responses.begin();
124  it != session.responses.end();
125  ++it) {
126  Response* response = it->second;
127  response->ready.notify_all();
128  }
129  }
130 }

150 void
151 ClientSession::Timer::handleTimerEvent()
152 {
153  std::unique_lock<std::mutex> mutexGuard(session.mutex);
154 
155  // Handle "spurious" wake-ups.
156  if (!session.messageSocket ||
157  session.numActiveRPCs == 0 ||
158  !session.errorMessage.empty()) {
159  return;
160  }
161 
162  // Send a ping or expire the session.
163  if (!session.activePing) {
164  VERBOSE("ClientSession is suspicious. Sending ping.");
165  session.activePing = true;
166  session.messageSocket->sendMessage(PING_MESSAGE_ID, Buffer());
167  schedule(TIMEOUT_MS * 1000 * 1000);
168  } else {
169  VERBOSE("ClientSession to %s timed out.",
170  session.address.toString().c_str());
171  // Fail all current and future RPCs.
172  session.errorMessage = ("Server " +
173  session.address.toString() +
174  " timed out");
175  // Notify any waiting RPCs.
176  for (auto it = session.responses.begin();
177  it != session.responses.end();
178  ++it) {
179  Response* response = it->second;
180  response->ready.notify_all();
181  }
182  }
183 }

187 ClientSession::ClientSession(Event::Loop& eventLoop,
188  const Address& address,
189  uint32_t maxMessageLength)
190  : self() // makeSession will fill this in shortly
191  , eventLoop(eventLoop)
192  , address(address)
193  , messageSocket()
194  , timer(*this)
195  , mutex()
196  , nextMessageId(1) // 0 is reserved for PING_MESSAGE_ID
197  , responses()
198  , errorMessage()
199  , numActiveRPCs(0)
200  , activePing(false)
201 {
202  int fd = socket(AF_INET, SOCK_STREAM, 0);
203  if (fd < 0) {
204  errorMessage = "Failed to create socket";
205  return;
206  }
207  int r = connect(fd,
208  address.getSockAddr(),
209  address.getSockAddrLen());
210  if (r != 0) {
211  errorMessage = "Failed to connect socket to " + address.toString();
212  close(fd);
213  return;
214  }
215  messageSocket.reset(new ClientMessageSocket(*this, fd, maxMessageLength));
216 }

237 OpaqueClientRPC
238 ClientSession::sendRequest(Buffer request)
239 {
240  MessageSocket::MessageId messageId;
241  {
242  std::unique_lock<std::mutex> mutexGuard(mutex);
243  messageId = nextMessageId;
244  ++nextMessageId;
245  responses[messageId] = new Response();
246 
247  ++numActiveRPCs;
248  if (numActiveRPCs == 1) {
249  // activePing's value was undefined while numActiveRPCs = 0
250  activePing = false;
251  timer.schedule(TIMEOUT_MS * 1000 * 1000);
252  }
253  }
254  // Release the mutex before sending so that receives can be processed
255  // simultaneously with sends.
256  if (messageSocket)
257  messageSocket->sendMessage(messageId, std::move(request));
258  OpaqueClientRPC rpc;
259  rpc.session = self.lock();
260  rpc.responseToken = messageId;
261  return rpc;
262 }

283 void
284 ClientSession::cancel(OpaqueClientRPC& rpc)
285 {
286  // The RPC may be holding the last reference to this session. This
287  // temporary reference makes sure this object isn't destroyed until after
288  // we return from this method. It must be the first line in this method.
289  std::shared_ptr<ClientSession> selfGuard(self.lock());
290 
291  // There are two ways to cancel an RPC:
292  // 1. If there's some thread currently blocked in wait(), this method marks
293  // the Response's status as CANCELED, and wait() will delete it later.
294  // 2. If there's no thread currently blocked in wait(), the Response is
295  // deleted entirely.
296  std::unique_lock<std::mutex> mutexGuard(mutex);
297  auto it = responses.find(rpc.responseToken);
298  if (it == responses.end())
299  return;
300  Response* response = it->second;
301  if (response->hasWaiter) {
302  response->status = Response::CANCELED;
303  response->ready.notify_all();
304  } else {
305  delete response;
306  responses.erase(it);
307  }
308 
309  --numActiveRPCs;
310  // Even if numActiveRPCs == 0, it's simpler here to just let the timer wake
311  // up an extra time and clean up. Otherwise, we'd need to grab an
312  // Event::Loop::Lock prior to the mutex to call deschedule() without
313  // inducing deadlock.
314 }

316 void
317 ClientSession::update(OpaqueClientRPC& rpc)
318 {
319  // The RPC may be holding the last reference to this session. This
320  // temporary reference makes sure this object isn't destroyed until after
321  // we return from this method. It must be the first line in this method.
322  std::shared_ptr<ClientSession> selfGuard(self.lock());
323 
324  std::unique_lock<std::mutex> mutexGuard(mutex);
325  auto it = responses.find(rpc.responseToken);
326  if (it == responses.end()) {
327  // RPC was cancelled, fields set already
328  assert(rpc.ready);
329  return;
330  }
331  Response* response = it->second;
332  if (response->status == Response::HAS_REPLY) {
333  rpc.reply = std::move(response->reply);
334  } else if (!errorMessage.empty()) {
335  rpc.errorMessage = errorMessage;
336  } else {
337  // If the RPC was canceled, then it'd be marked ready and update()
338  // wouldn't be called again.
339  assert(response->status != Response::CANCELED);
340  return; // not ready
341  }
342  rpc.ready = true;
343  rpc.session.reset();
344 
345  delete response;
346  responses.erase(it);
347 }

349 void
350 ClientSession::wait(const OpaqueClientRPC& rpc)
351 {
352  // The RPC may be holding the last reference to this session. This
353  // temporary reference makes sure this object isn't destroyed until after
354  // we return from this method. It must be the first line in this method.
355  std::shared_ptr<ClientSession> selfGuard(self.lock());
356 
357  std::unique_lock<std::mutex> mutexGuard(mutex);
358  while (true) {
359  auto it = responses.find(rpc.responseToken);
360  if (it == responses.end())
361  return; // RPC was cancelled or already updated
362  Response* response = it->second;
363  if (response->status == Response::HAS_REPLY) {
364  return; // RPC has completed
365  } else if (response->status == Response::CANCELED) {
366  // RPC was cancelled, finish cleaning up
367  delete response;
368  responses.erase(it);
369  return;
370  } else if (!errorMessage.empty()) {
371  return; // session has error
372  }
373  response->hasWaiter = true;
374  response->ready.wait(mutexGuard);
375  response->hasWaiter = false;
376  }
377 }
