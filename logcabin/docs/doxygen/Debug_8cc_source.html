84 /**
85  * Convert a string to a log level.
86  * PANICs if the string is not a valid log level (case insensitive).
87  */
88 LogLevel
89 logLevelFromString(const std::string& level)
90 {
91  for (uint32_t i = 0; logLevelToString[i] != NULL; ++i) {
92  if (strcasecmp(logLevelToString[i], level.c_str()) == 0)
93  return LogLevel(i);
94  }
95  log((LogLevel::ERROR), __FILE__, __LINE__, __FUNCTION__,
96  "'%s' is not a valid log level.\n", level.c_str());
97  abort();
98 }

100 /**
101  * From the policy, calculate the most verbose log level that should be
102  * displayed for this file.
103  * This is slow, so isLogging() caches the results in isLoggingCache.
104  *
105  * Must be called with #mutex held.
106  *
107  * \param fileName
108  * Relative filename.
109  */
110 LogLevel
111 getLogLevel(const char* fileName)
112 {
113  for (auto it = policy.begin(); it != policy.end(); ++it) {
114  const std::string& pattern = it->first;
115  const std::string& logLevel = it->second;
116  if (Core::StringUtil::startsWith(fileName, pattern) ||
117  Core::StringUtil::endsWith(fileName, pattern)) {
118  return logLevelFromString(logLevel);
119  }
120  }
121  return LogLevel::NOTICE;
122 }

141 /**
142  * Strip out the common prefix of a filename to get a path from the project's
143  * root directory.
144  * \param fileName
145  * An absolute or relative filename, usually the value of __FILE__.
146  * \return
147  * A nicer way to show display 'fileName' to the user.
148  * For example, this file would yield "Core/Debug.cc".
149  */
150 const char*
151 relativeFileName(const char* fileName)
152 {
153  // Remove the prefix only if it matches that of __FILE__. This check is
154  // needed in case someone compiles different files using different paths.
155  if (strncmp(fileName, __FILE__, lengthFilePrefix) == 0)
156  return fileName + lengthFilePrefix;
157  else
158  return fileName;
159 }

188 bool
189 isLogging(LogLevel level, const char* fileName)
190 {
191  std::unique_lock<std::mutex> lockGuard(mutex);
192  LogLevel verbosity;
193  auto it = isLoggingCache.find(fileName);
194  if (it == isLoggingCache.end()) {
195  verbosity = getLogLevel(relativeFileName(fileName));
196  isLoggingCache[fileName] = verbosity;
197  } else {
198  verbosity = it->second;
199  }
200  return uint32_t(level) <= uint32_t(verbosity);
201 }

203 void
204 log(LogLevel level,
205  const char* fileName, uint32_t lineNum, const char* functionName,
206  const char* format, ...)
207 {
208  va_list ap;
209  struct timespec now;
210  clock_gettime(CLOCK_REALTIME, &now);
211 
212  // This ensures that output on stderr won't be interspersed with other
213  // output. This normally happens automatically for a single call to
214  // fprintf, but must be explicit since we're using two calls here.
215  flockfile(stream);
216 
217  fprintf(stream, "%010lu.%06lu %s:%d in %s() %s[%s:%s]: ",
218  now.tv_sec, now.tv_nsec / 1000,
219  relativeFileName(fileName), lineNum, functionName,
220  logLevelToString[uint32_t(level)],
221  processName.c_str(), ThreadId::getName().c_str());
222 
223  va_start(ap, format);
224  vfprintf(stream, format, ap);
225  va_end(ap);
226 
227  funlockfile(stream);
228 
229  fflush(stream);
230 }
