47 MemoryLog::append(const std::vector<const Entry*>& newEntries)
48 {
49  uint64_t firstIndex = startIndex + entries.size();
50  uint64_t lastIndex = firstIndex + newEntries.size() - 1;

95 void
96 MemoryLog::truncatePrefix(uint64_t firstIndex)
97 {
98  if (firstIndex > startIndex) {
99  // Erase log entries in range [startIndex, firstIndex), so deque
100  // offsets in range [0, firstIndex - startIndex). Be careful not to
101  // erase past the end of the deque (STL doesn't check for this).
102  entries.erase(entries.begin(),
103  entries.begin() + std::min(firstIndex - startIndex,
104  entries.size()));
105  startIndex = firstIndex;
106  }
107 }

109 void
110 MemoryLog::truncateSuffix(uint64_t lastIndex)
111 {
112  if (lastIndex < startIndex)
113  entries.clear();
114  else if (lastIndex < startIndex - 1 + entries.size())
115  entries.resize(lastIndex - startIndex + 1);
116 }
