28 /**
29  * A wrapper around std::mutex that is useful for testing purposes. You can set
30  * a callback to be called when the mutex is locked and before it is unlocked.
31  * This callback can, for example, check the invariants on the protected state.
32  *
33  * The interface to this class is the same as std::mutex.
34  */
35 class Mutex {
36  public:
37  typedef std::mutex::native_handle_type native_handle_type;
38 
39  Mutex()
40  : m()
41  , callback()
42  {
43  }
44 
45  void
46  lock() {
47  m.lock();
48  if (callback)
49  callback();
50  }
51 
52  bool
53  try_lock() {
54  bool l = m.try_lock();
55  if (l) {
56  if (callback)
57  callback();
58  }
59  return l;
60  }
61 
62  void
63  unlock() {
64  // TODO(ongardie): apparently try_lock->false...unlock is allowed, but
65  // this will then call the callback without the lock, which is unsafe.
66  if (callback)
67  callback();
68  m.unlock();
69  }
70 
71  native_handle_type
72  native_handle() {
73  return m.native_handle();
74  }
75 
76  private:
77  /// Underlying mutex.
78  std::mutex m;
79  public:
80  /**
81  * This function will be called with the lock held after the lock is
82  * acquired and before it is released.
83  */
84  std::function<void()> callback;
85 
86  friend class ConditionVariable;
87 };
