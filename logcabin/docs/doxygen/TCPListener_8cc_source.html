67 std::string
68 TCPListener::bind(const Address& listenAddress)
69 {
70  using Core::StringUtil::format;
71 
72  if (!listenAddress.isValid()) {
73  return format("Can't listen on invalid address: %s",
74  listenAddress.toString().c_str());
75  }
76 
77  int fd = socket(AF_INET, SOCK_STREAM|SOCK_CLOEXEC, 0);
78  if (fd < 0)
79  PANIC("Could not create new TCP socket");
80 
81  int flag = 1;
82  int r = setsockopt(fd, SOL_SOCKET, SO_REUSEADDR,
83  &flag, sizeof(flag));
84  if (r < 0) {
85  PANIC("Could not set SO_REUSEADDR on socket: %s",
86  strerror(errno));
87  }
88 
89 
90  r = ::bind(fd, listenAddress.getSockAddr(),
91  listenAddress.getSockAddrLen());
92  if (r != 0) {
93  std::string msg =
94  format("Could not bind to address %s: %s%s",
95  listenAddress.toString().c_str(),
96  strerror(errno),
97  errno == EINVAL ? " (is the port in use?)" : "");
98  r = close(fd);
99  if (r != 0) {
100  WARNING("Could not close socket that failed to bind: %s",
101  strerror(errno));
102  }
103  return msg;
104  }
105 
106  // Why 128? No clue. It's what libevent was setting it to.
107  r = listen(fd, 128);
108  if (r != 0) {
109  PANIC("Could not invoke listen() on address %s: %s",
110  listenAddress.toString().c_str(),
111  strerror(errno));
112  }
113 
114  std::unique_lock<Core::Mutex> lock(mutex);
115  boundListeners.emplace_back(*this, eventLoop, fd);
116  return "";
117 }
