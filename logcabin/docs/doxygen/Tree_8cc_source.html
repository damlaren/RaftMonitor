197 Directory::dumpSnapshot(google::protobuf::io::CodedOutputStream& stream) const
198 {
199  // create protobuf of this dir, listing all children
200  Snapshot::Directory dir;
201  for (auto it = directories.begin(); it != directories.end(); ++it)
202  dir.add_directories(it->first);
203  for (auto it = files.begin(); it != files.end(); ++it)
204  dir.add_files(it->first);
205 
206  // write dir into stream
207  int size = dir.ByteSize();
208  stream.WriteLittleEndian32(size);
209  dir.SerializeWithCachedSizes(&stream);
210 
211  // dump children in the same order
212  for (auto it = directories.begin(); it != directories.end(); ++it)
213  it->second.dumpSnapshot(stream);
214  for (auto it = files.begin(); it != files.end(); ++it)
215  it->second.dumpSnapshot(stream);

249 Path::Path(const std::string& symbolic)
250  : result()
251  , symbolic(symbolic)
252  , parents()
253  , target()
254 {
255  if (!Core::StringUtil::startsWith(symbolic, "/")) {
256  result.status = Status::INVALID_ARGUMENT;
257  result.error = format("'%s' is not a valid path",
258  symbolic.c_str());
259  return;
260  }
261 
262  // Add /root prefix (see docs for Tree::superRoot)
263  parents.push_back("root");
264 
265  // Split the path into a list of parent components and a target.
266  std::string word;
267  for (auto it = symbolic.begin(); it != symbolic.end(); ++it) {
268  if (*it == '/') {
269  if (!word.empty()) {
270  parents.push_back(word);
271  word.clear();
272  }
273  } else {
274  word += *it;
275  }
276  }
277  if (!word.empty())
278  parents.push_back(word);
279  target = parents.back();
280  parents.pop_back();
281 }

283 std::string
284 Path::parentsThrough(std::vector<std::string>::const_iterator end) const
285 {
286  auto it = parents.begin();
287  ++it; // skip "root"
288  ++end; // end was inclusive, now exclusive
289  if (it == end)
290  return "/";
291  std::string ret;
292  do {
293  ret += "/" + *it;
294  ++it;
295  } while (it != end);
296  return ret;
297 }

318 Result
319 Tree::normalLookup(const Path& path, const Directory** parent) const
320 {
321  *parent = NULL;
322  Result result;
323  const Directory* current = &superRoot;
324  for (auto it = path.parents.begin(); it != path.parents.end(); ++it) {
325  const Directory* next = current->lookupDirectory(*it);
326  if (next == NULL) {
327  if (current->lookupFile(*it) == NULL) {
328  result.status = Status::LOOKUP_ERROR;
329  result.error = format("Parent %s of %s does not exist",
330  path.parentsThrough(it).c_str(),
331  path.symbolic.c_str());
332  } else {
333  result.status = Status::TYPE_ERROR;
334  result.error = format("Parent %s of %s is a file",
335  path.parentsThrough(it).c_str(),
336  path.symbolic.c_str());
337  }
338  return result;
339  }
340  current = next;
341  }
342  *parent = current;
343  return result;
344 }

387 {

457 Result
458 Tree::removeDirectory(const std::string& symbolicPath)
459 {
460  Path path(symbolicPath);
461  if (path.result.status != Status::OK)
462  return path.result;
463  Directory* parent;
464  Result result = normalLookup(path, &parent);
465  if (result.status == Status::LOOKUP_ERROR) {
466  // no parent, already done
467  return Result();
468  }
469  if (result.status != Status::OK)
470  return result;
471  Directory* targetDir = parent->lookupDirectory(path.target);
472  if (targetDir == NULL) {
473  if (parent->lookupFile(path.target)) {
474  result.status = Status::TYPE_ERROR;
475  result.error = format("%s is a file",
476  path.symbolic.c_str());
477  return result;
478  } else {
479  // target does not exist, already done
480  return result;
481  }
482  }
483  parent->removeDirectory(path.target);
484  if (parent == &superRoot) { // removeDirectory("/")
485  // If the caller is trying to remove the root directory, we remove the
486  // contents but not the directory itself. The easiest way to do this
487  // is to drop but then recreate the directory.
488  parent->makeDirectory(path.target);
489  }
490  return result;
491 }
