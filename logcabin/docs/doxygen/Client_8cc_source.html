174 Result
175 Tree::setWorkingDirectory(const std::string& newWorkingDirectory)
176 {
177  // This method sets the working directory regardless of whether it
178  // succeeds -- that way if it doesn't, future relative paths on this Tree
179  // will result in errors instead of operating on the prior working
180  // directory.
181 
182  std::unique_lock<std::mutex> lockGuard(mutex);
183  std::string realPath;
184  Result result = treeDetails->clientImpl->canonicalize(
185  newWorkingDirectory,
186  treeDetails->workingDirectory,
187  realPath);
188  std::shared_ptr<TreeDetails> newTreeDetails(new TreeDetails(*treeDetails));
189  if (result.status != Status::OK) {
190  newTreeDetails->workingDirectory = Core::StringUtil::format(
191  "invalid from prior call to setWorkingDirectory('%s') "
192  "relative to '%s'",
193  newWorkingDirectory.c_str(),
194  treeDetails->workingDirectory.c_str());
195  treeDetails = newTreeDetails;
196  return result;
197  }
198  newTreeDetails->workingDirectory = realPath;
199  treeDetails = newTreeDetails;
200  return treeDetails->clientImpl->makeDirectory(realPath, "",
201  treeDetails->condition);
202 }

217 Result
218 Tree::setCondition(const std::string& path, const std::string& value)
219 {
220  // This method sets the condition regardless of whether it succeeds -- that
221  // way if it doesn't, future calls on this Tree will result in errors
222  // instead of operating on the prior condition.
223 
224  std::unique_lock<std::mutex> lockGuard(mutex);
225  std::string realPath;
226  Result result = treeDetails->clientImpl->canonicalize(
227  path,
228  treeDetails->workingDirectory,
229  realPath);
230  std::shared_ptr<TreeDetails> newTreeDetails(new TreeDetails(*treeDetails));
231  if (result.status != Status::OK) {
232  newTreeDetails->condition = {
233  Core::StringUtil::format(
234  "invalid from prior call to setCondition('%s') "
235  "relative to '%s'",
236  path.c_str(),
237  treeDetails->workingDirectory.c_str()),
238  value,
239  };
240  treeDetails = newTreeDetails;
241  return result;
242  }
243  newTreeDetails->condition = {path, value};
244  treeDetails = newTreeDetails;
245  return Result();
246 }
