89 ClientRPC::Status
90 ClientRPC::waitForReply(google::protobuf::Message* response,
91  google::protobuf::Message* serviceSpecificError)
92 {
93  opaqueRPC.waitForReply();
94  std::string error = opaqueRPC.getErrorMessage();
95  if (!error.empty())
96  return Status::RPC_FAILED;
97  const Buffer& responseBuffer = *opaqueRPC.peekReply();
98 
99  // Extract the response's status field.
100  if (responseBuffer.getLength() < sizeof(ResponseHeaderPrefix)) {
101  PANIC("The response from the server was too short to be valid. "
102  "This probably indicates network or memory corruption.");
103  }
104  ResponseHeaderPrefix responseHeaderPrefix =
105  *static_cast<const ResponseHeaderPrefix*>(responseBuffer.getData());
106  responseHeaderPrefix.fromBigEndian();
107  if (responseHeaderPrefix.status == ProtocolStatus::INVALID_VERSION) {
108  // The server doesn't understand this version of the header
109  // protocol. Since this library only runs version 1 of the
110  // protocol, this shouldn't happen if servers continue supporting
111  // version 1.
112  PANIC("This client is too old to talk to the server. "
113  "You'll need to update your client library.");
114  }
115 
116  if (responseBuffer.getLength() < sizeof(ResponseHeaderVersion1)) {
117  PANIC("The response from the server was too short to be valid. "
118  "This probably indicates network or memory corruption.");
119  }
120  ResponseHeaderVersion1 responseHeader =
121  *static_cast<const ResponseHeaderVersion1*>(responseBuffer.getData());
122  responseHeader.fromBigEndian();
123 
124  switch (responseHeader.prefix.status) {
125 
126  // The RPC succeeded. Parse the response into a protocol buffer.
127  case ProtocolStatus::OK:
128  if (response != NULL &&
129  !RPC::ProtoBuf::parse(responseBuffer, *response,
130  sizeof(responseHeader))) {
131  PANIC("Could not parse the protocol buffer out of the server "
132  "response");
133  }
134  return Status::OK;
135 
136  // The RPC failed in a service-specific way. Parse the response into a
137  // protocol buffer.
138  case ProtocolStatus::SERVICE_SPECIFIC_ERROR:
139  if (serviceSpecificError != NULL &&
140  !RPC::ProtoBuf::parse(responseBuffer, *serviceSpecificError,
141  sizeof(responseHeader))) {
142  PANIC("Could not parse the protocol buffer out of the "
143  "service-specific error details");
144  }
145  return Status::SERVICE_SPECIFIC_ERROR;
146 
147  // The server does not have the requested service.
148  case ProtocolStatus::INVALID_SERVICE:
149  PANIC("The server is not running the requested service.");
150 
151  // The server disliked our request. This shouldn't happen because
152  // the higher layers of software were supposed to negotiate an RPC
153  // protocol version.
154  case ProtocolStatus::INVALID_REQUEST:
155  PANIC("The server found the request to be invalid. This "
156  "indicates a bug in the client or server in negotiating "
157  "which RPCs the client may legally send to the server.");
158 
159  default:
160  // The server shouldn't reply back with status codes we don't
161  // understand. That's why we gave it a version number in the
162  // request header.
163  PANIC("Unknown status %u returned from server after sending it "
164  "protocol version 1 in the request header. This probably "
165  "indicates a bug in the server.",
166  uint32_t(responseHeader.prefix.status));
167  }
168 
169 }
