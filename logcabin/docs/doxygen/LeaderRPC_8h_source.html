37 /**
38  * This class is used to send RPCs from clients to the leader of the LogCabin
39  * cluster. It automatically finds and connects to the leader and transparently
40  * rolls over to a new leader when necessary.
41  *
42  * There are two implementations of this interface: LeaderRPC is probably the
43  * one you're interested in. LeaderRPCMock is used for unit testing only.
44  */
45 class LeaderRPCBase {
46  public:
47  /**
48  * RPC operation code.
49  */
50  typedef Protocol::Client::OpCode OpCode;
51 
52  /// Constructor.
53  LeaderRPCBase() {}
54 
55  /// Destructor.
56  virtual ~LeaderRPCBase() {}
57 
58  /**
59  * Execute an RPC on the cluster leader.
60  * This class guarantees that the RPC will be executed at least once.
61  * \param opCode
62  * RPC operation code. The caller must guarantee that this is a valid
63  * opCode. (If the server rejects it, this will PANIC.)
64  * \param request
65  * The parameters for the operation. The caller must guarantee that
66  * this is a well-formed request. (If the server rejects it, this will
67  * PANIC.)
68  * \param[out] response
69  * The response to the operation will be filled in here.
70  */
71  virtual void call(OpCode opCode,
72  const google::protobuf::Message& request,
73  google::protobuf::Message& response) = 0;
74 
75  // LeaderRPCBase is not copyable
76  LeaderRPCBase(const LeaderRPCBase&) = delete;
77  LeaderRPCBase& operator=(const LeaderRPCBase&) = delete;
78 };
