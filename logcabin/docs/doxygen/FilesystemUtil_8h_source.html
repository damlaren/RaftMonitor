32 /**
33  * A File object is just a wrapper around a file descriptor; it represents
34  * either an open file, an open directory, or an empty placeholder. It takes
35  * charge of closing the file descriptor when it is done and tracks the path
36  * used to open the file descriptor in order to provide useful error messages.
37  */
38 class File {
39  public:
40  /// Default constructor.
41  File();
42  /// Move constructor.
43  File(File&& other);
44  /**
45  * Constructor.
46  * \param fd
47  * An open file descriptor.
48  * \param path
49  * The path used to open fd; used for error messages.
50  */
51  File(int fd, std::string path);
52  /// Destructor.
53  ~File();
54 
55  /// Move assignment.
56  File& operator=(File&& other);
57 
58  /**
59  * Close the file.
60  * This object's fd and path are cleared.
61  */
62  void close();
63 
64  /**
65  * Disassociate the file descriptor from this object.
66  * The caller is in charge of closing the file descriptor. This object's fd
67  * and path are cleared.
68  * \return
69  * The file descriptor.
70  */
71  int release();
72 
73  /**
74  * The open file descriptor, or -1 otherwise.
75  */
76  int fd;
77  /**
78  * The path used to open fd, or empty. Used for error messages.
79  */
80  std::string path;
81 
82  // File is not copyable.
83  File(const File&) = delete;
84  File& operator=(const File&) = delete;
85 };

266  * Copy some number of bytes of the file into a user-supplied buffer.

268  * If there are not enough bytes in the file, this will PANIC. See

271  * \param offset

280  /**
281  * Copy up to some number of bytes of the file into a user-supplied buffer.
282  *
283  * \param offset
284  * The number of bytes into the file at which to start copying.
285  * \param[out] buf
286  * The destination buffer to copy into.
287  * \param maxLength
288  * The maximum number of bytes to copy.
289  * \return
290  * The number of bytes copied. This can be fewer than maxLength if the
291  * file ended first.
292  */
293  uint64_t copyPartial(uint64_t offset, void* buf, uint64_t maxLength);
294 
295  /**
296  * Get a pointer to a region of the file.
297  *
298  * If there are not enough bytes in the file, this will PANIC.
299  *
300  * \tparam T
301  * The return type is casted to a pointer of T.
302  * \param offset
303  * The number of bytes into the file at which to return the pointer.
304  * \param length
305  * The number of bytes that must be valid after the offset.
306  * \return
307  * A pointer to a buffer containing the 'length' bytes starting at
308  * 'offset' in the file. The caller may not modify this buffer. The
309  * returned buffer will remain valid until this object is destroyed.
310  */
311  template<typename T = void>
312  const T*
313  get(uint64_t offset, uint64_t length) {
314  return reinterpret_cast<const T*>(getHelper(offset, length));
315  }
316 
317  private:
318  /// Used internally by get().
319  const void* getHelper(uint64_t offset, uint64_t length);
320  /// An open file descriptor for the file to read.
321  File file;
322  /// The number of bytes in the file.
323  uint64_t fileLen;
324  /// The value returned by mmap(), or NULL for empty files.
325  const void* map;
326  FileContents(const FileContents&) = delete;
327  FileContents& operator=(const FileContents&) = delete;
328 };
