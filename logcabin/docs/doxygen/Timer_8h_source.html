25 /**
26  * A Timer is called by the Event::Loop when time has elapsed.
27  * The client should inherit from this and implement the trigger method for
28  * when the timer expires.
29  *
30  * Timers can be added and scheduled from any thread, but they will always fire
31  * on the thread running the Event::Loop.
32  */
33 class Timer : public File {
34  public:
35 
36  /**
37  * Construct a timer but do not schedule it to trigger.
38  * It will not fire until schedule() is invoked.
39  * \param eventLoop
40  * Event::Loop that will manage this timer.
41  */
42  explicit Timer(Event::Loop& eventLoop);

54  /**
55  * This method is overridden by a subclass and invoked when the timer
56  * expires. This method will be invoked by the main event loop on whatever
57  * thread is running the Event::Loop.
58  */
59  virtual void handleTimerEvent() = 0;
60 
61  /**
62  * Start the timer.
63  * \param nanoseconds
64  * The timer will trigger once this number of nanoseconds have elapsed.
65  * If the timer was already scheduled, the old time is forgotten.
66  */
67  void schedule(uint64_t nanoseconds);
68 
69  /**
70  * Stop the timer from calling handleTimerEvent().
71  * This will cancel the current timer, if any, so that handleTimerEvent()
72  * is not called until the next time it is scheduled.
73  *
74  * This method behaves in a friendly way when called concurrently with the
75  * timer firing:
76  * - If this method is called from another thread and handleTimerEvent() is
77  * running concurrently on the event loop thread, deschedule() will wait
78  * for handleTimerEvent() to complete before returning.
79  * - If this method is called from the event loop thread and
80  * handleTimerEvent() is currently being fired, then deschedule() can't
81  * wait and returns immediately.
82  */
83  void deschedule();
84 
85  /**
86  * Returns true if the timer has been scheduled and has not yet fired.
87  * This is prone to races; it's primarily useful for unit tests.
88  * \return
89  * True if the timer has been scheduled and will eventually call
90  * handleTimerEvent().
91  * False if the timer is currently running handleTimerEvent() or if it
92  * is not scheduled to call handleTimerEvent() in the future.
93  */
94  bool isScheduled() const;
95 
96  private:
97  // Timer is not copyable.
98  Timer(const Timer&) = delete;
99  Timer& operator=(const Timer&) = delete;
