49 void
50 ServiceMock::handleRPC(RPC::ServerRPC rpc)
51 {
52  if (responseQueue.empty()) {
53  rpc.rejectInvalidRequest();
54  return;
55  }
56  Expected& exp = responseQueue.front();
57  EXPECT_EQ(exp.opCode, rpc.getOpCode());
58  if (rpc.getOpCode() != exp.opCode) {
59  rpc.rejectInvalidRequest();
60  return;
61  }
62  std::unique_ptr<Message> actualReq(exp.request->New());
63  bool b = rpc.getRequest(*actualReq);
64  EXPECT_TRUE(b) << "rpc.getRequest() failed";
65  if (!b)
66  return;
67  EXPECT_EQ(*exp.request, *actualReq);
68  if (*exp.request != *actualReq) {
69  rpc.rejectInvalidRequest();
70  return;
71  }
72  exp.response->handleRPC(std::move(rpc));
73  responseQueue.pop();
74 }
