48 Loop::Lock::Lock(Event::Loop& eventLoop)
49  : eventLoop(eventLoop)
50 {
51  std::unique_lock<std::mutex> lockGuard(eventLoop.mutex);
52  ++eventLoop.numLocks;
53  if (eventLoop.runningThread != Core::ThreadId::getId() &&
54  eventLoop.lockOwner != Core::ThreadId::getId()) {
55  // This is an actual lock: we're not running inside the event loop, and
56  // we're not recursively locking.
57  if (eventLoop.runningThread != Core::ThreadId::NONE)
58  eventLoop.breakTimer->schedule(0);
59  while (eventLoop.runningThread != Core::ThreadId::NONE ||
60  eventLoop.lockOwner != Core::ThreadId::NONE) {
61  eventLoop.safeToLock.wait(lockGuard);
62  }
63  // Take ownership of the lock
64  eventLoop.lockOwner = Core::ThreadId::getId();
65  }
66  ++eventLoop.numActiveLocks;
67 }

69 Loop::Lock::~Lock()
70 {
71  std::unique_lock<std::mutex> lockGuard(eventLoop.mutex);
72  --eventLoop.numLocks;
73  --eventLoop.numActiveLocks;
74  if (eventLoop.numActiveLocks == 0) {
75  eventLoop.lockOwner = Core::ThreadId::NONE;
76  if (eventLoop.numLocks == 0)
77  eventLoop.unlocked.notify_one();
78  else
79  eventLoop.safeToLock.notify_one();
80  }
81 }

115 void
116 Loop::runForever()
117 {
118  while (true) {
119  { // Handle Loop::Lock requests and exiting.
120  std::unique_lock<std::mutex> lockGuard(mutex);
121  runningThread = Core::ThreadId::NONE;
122  // Wait for all Locks to finish up
123  while (numLocks > 0) {
124  safeToLock.notify_one();
125  unlocked.wait(lockGuard);
126  }
127  if (shouldExit) {
128  shouldExit = false;
129  return;
130  }
131  runningThread = Core::ThreadId::getId();
132  }
133 
134  // Block in the kernel for events, then process them.
135  // TODO(ongaro): It'd be more efficient to handle more than 1 event at
136  // a time, but that complicates the interface: if a handler removes
137  // itself from the poll set and deletes itself, we don't want further
138  // events to call that same handler. For example, if a socket is
139  // dup()ed so that the receive side handles events separately from the
140  // send side, both are active events, and the first deletes the object,
141  // this could cause trouble.
142  enum { NUM_EVENTS = 1 };
143  struct epoll_event events[NUM_EVENTS];
144  int r = epoll_wait(epollfd, events, NUM_EVENTS, -1);
145  if (r <= 0) {
146  if (errno == EINTR) // caused by GDB
147  continue;
148  PANIC("epoll_wait failed: %s", strerror(errno));
149  }
150  for (int i = 0; i < r; ++i) {
151  Event::File& file = *static_cast<Event::File*>(events[i].data.ptr);
152  file.handleFileEvent(events[i].events);
153  }
154  }
155 }
