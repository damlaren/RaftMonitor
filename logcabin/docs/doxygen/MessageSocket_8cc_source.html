183 void
184 MessageSocket::sendMessage(MessageId messageId, Buffer contents)
185 {
186  // Check the message length.
187  if (contents.getLength() > maxMessageLength) {
188  PANIC("Message of length %u bytes is too long to send "
189  "(limit is %u bytes)",
190  contents.getLength(), maxMessageLength);
191  }
192 
193  bool kick;
194  { // Place the message on the outbound queue.
195  std::lock_guard<Core::Mutex> lock(outboundQueueMutex);
196  kick = outboundQueue.empty();
197  outboundQueue.emplace_back(messageId, std::move(contents));
198  }
199  // Make sure the SendSocket is set up to call writable().
200  if (kick)
201  sendSocket.setEvents(EPOLLOUT|EPOLLONESHOT);
202 }

218 void
219 MessageSocket::readable()
220 {
221  if (receiveSocket.fd < 0)
222  return;
223  // Try to read data from the kernel until there is no more left.
224  while (true) {
225  if (inbound.bytesRead < sizeof(Header)) {
226  // Receiving header
227  ssize_t bytesRead = read(
228  reinterpret_cast<char*>(&inbound.header) + inbound.bytesRead,
229  sizeof(Header) - inbound.bytesRead);
230  if (bytesRead == -1) {
231  disconnect();
232  return;
233  }
234  inbound.bytesRead += bytesRead;
235  if (inbound.bytesRead < sizeof(Header))
236  return;
237  // Transition to receiving data
238  inbound.header.fromBigEndian();
239  if (inbound.header.payloadLength > maxMessageLength) {
240  WARNING("Dropping message that is too long to receive "
241  "(message is %u bytes, limit is %u bytes)",
242  inbound.header.payloadLength, maxMessageLength);
243  disconnect();
244  return;
245  }
246  inbound.message.setData(new char[inbound.header.payloadLength],
247  inbound.header.payloadLength,
248  Buffer::deleteArrayFn<char>);
249  }
250  // Don't use 'else' here; we want to check this branch for two reasons:
251  // First, if there is a header with a length of 0, the socket won't be
252  // readable, but we still need to process the message. Second, most of
253  // the time the header will arrive with at least some data. It makes
254  // sense to go ahead and try a non-blocking read, rather than going
255  // back to the event loop.
256  if (inbound.bytesRead >= sizeof(Header)) {
257  // Receiving data
258  size_t payloadBytesRead = inbound.bytesRead - sizeof(Header);
259  ssize_t bytesRead = read(
260  (static_cast<char*>(inbound.message.getData()) +
261  payloadBytesRead),
262  inbound.header.payloadLength - payloadBytesRead);
263  if (bytesRead == -1) {
264  disconnect();
265  return;
266  }
267  inbound.bytesRead += bytesRead;
268  if (inbound.bytesRead < (sizeof(Header) +
269  inbound.header.payloadLength)) {
270  return;
271  }
272  onReceiveMessage(inbound.header.messageId,
273  std::move(inbound.message));
274  // Transition to receiving header
275  inbound.bytesRead = 0;
276  }
277  }

281 MessageSocket::read(void* buf, size_t maxBytes)
282 {
283  ssize_t actual = recv(receiveSocket.fd, buf, maxBytes, MSG_DONTWAIT);
284  if (actual > 0)
285  return actual;
286  if (actual == 0 || // peer performed orderly shutdown.
287  errno == ECONNRESET || errno == ETIMEDOUT) {
288  return -1;
289  }
290  if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR)
291  return 0;
292  PANIC("Error while reading from socket: %s", strerror(errno));

295 void
296 MessageSocket::writable()
297 {
298  if (receiveSocket.fd < 0)
299  return;
300  // Each iteration of this loop tries to write one message
301  // from outboundQueue.
302  while (true) {
303 
304  // Get the next outbound message.
305  Outbound outbound;
306  int flags = MSG_DONTWAIT | MSG_NOSIGNAL;
307  {
308  std::lock_guard<Core::Mutex> lock(outboundQueueMutex);
309  if (outboundQueue.empty())
310  return;
311  outbound = std::move(outboundQueue.front());
312  outboundQueue.pop_front();
313  if (!outboundQueue.empty())
314  flags |= MSG_MORE;
315  }
316 
317  // Use an iovec to send everything in one kernel call: one iov for the
318  // header, another for the payload.
319  enum { IOV_LEN = 2 };
320  struct iovec iov[IOV_LEN];
321  iov[0].iov_base = &outbound.header;
322  iov[0].iov_len = sizeof(Header);
323  iov[1].iov_base = outbound.message.getData();
324  iov[1].iov_len = outbound.message.getLength();
325 
326  { // Skip the parts of the iovec that have already been sent.
327  size_t bytesSent = outbound.bytesSent;
328  for (uint32_t i = 0; i < IOV_LEN; ++i) {
329  iov[i].iov_base = (static_cast<char*>(iov[i].iov_base) +
330  bytesSent);
331  if (bytesSent < iov[i].iov_len) {
332  iov[i].iov_len -= bytesSent;
333  bytesSent = 0;
334  break;
335  } else {
336  bytesSent -= iov[i].iov_len;
337  iov[i].iov_len = 0;
338  }
339  }
340  }
341 
342  struct msghdr msg;
343  memset(&msg, 0, sizeof(msg));
344  msg.msg_iov = iov;
345  msg.msg_iovlen = IOV_LEN;
346 
347  // Do the actual send
348  ssize_t bytesSent = sendmsg(sendSocket.fd, &msg, flags);
349  if (bytesSent < 0) {
350  if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
351  // Wasn't able to send, try again later.
352  bytesSent = 0;
353  } else if (errno == ECONNRESET || errno == EPIPE) {
354  // Connection closed; disconnect this end.
355  // This must be the last line to touch this object, in case
356  // onDisconnect() deletes this object.
357  disconnect();
358  return;
359  } else {
360  // Unexpected error.
361  PANIC("Error while writing to socket %d: %s",
362  sendSocket.fd, strerror(errno));
363  }
364  }
365 
366  // Sent successfully.
367  outbound.bytesSent += bytesSent;
368  if (outbound.bytesSent != (sizeof(Header) +
369  outbound.message.getLength())) {
370  sendSocket.setEvents(EPOLLOUT|EPOLLONESHOT);
371  std::unique_lock<Core::Mutex> lockGuard(outboundQueueMutex);
372  outboundQueue.emplace_front(std::move(outbound));
373  return;
374  }
375  }
376 }
