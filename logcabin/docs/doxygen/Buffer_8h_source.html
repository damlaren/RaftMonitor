25 /**
26  * A container for opaque data. This makes it easy for code to operate on data
27  * without having to explicitly pass around its length and whether and how to
28  * free its memory.
29  */
30 class Buffer {
31  public:
32  /**
33  * A deleter is a function that can free the memory contained in the
34  * Buffer. Examples include C's free() function for memory allocated with
35  * malloc(), and deleteObjectFn() and deleteArrayFn() defined in this
36  * class.
37  * \param data
38  * The memory that should be reclaimed.
39  */
40  typedef void (*Deleter)(void* data);

42  /**
43  * A Deleter that uses C++'s delete keyword.
44  * This should be used with data that was allocated with C++'s new keyword.
45  * For new[] and delete[], see deleteArrayFn.
46  * \param data
47  * The memory that should be reclaimed.
48  * \tparam T
49  * The type of the object as it was allocated with new.
50  */
51  template<typename T>
52  static void deleteObjectFn(void* data) {
53  delete static_cast<T>(data);
54  }
55 
56  /**
57  * A Deleter that uses C++'s array delete[] keyword.
58  * This should be used with data that was allocated with C++'s array new[]
59  * keyword. For new and delete on individual objects, see deleteArray.
60  * \param data
61  * The memory that should be reclaimed.
62  * \tparam T
63  * The type of each element of the array as it was allocated with
64  * new[].
65  */
66  template<typename T>
67  static void deleteArrayFn(void* data) {
68  delete[] static_cast<T*>(data);
69  }
