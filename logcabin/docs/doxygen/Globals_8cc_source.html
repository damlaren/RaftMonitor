67 void
68 Globals::init(uint64_t serverId)
69 {
70  if (!raft) {
71  raft.reset(new RaftConsensus(*this));
72  }
73 
74  if (!raftService) {
75  raftService.reset(new RaftService(*this));
76  }
77 
78  if (!clientService) {
79  clientService.reset(new ClientService(*this));
80  }
81 
82  if (!rpcServer) {
83  rpcServer.reset(new RPC::Server(eventLoop,
84  Protocol::Common::MAX_MESSAGE_LENGTH));
85 
86  uint32_t maxThreads = config.read<uint16_t>("maxThreads", 16);
87  rpcServer->registerService(Protocol::Common::ServiceId::RAFT_SERVICE,
88  raftService,
89  maxThreads);
90  rpcServer->registerService(Protocol::Common::ServiceId::CLIENT_SERVICE,
91  clientService,
92  maxThreads);
93 
94 
95  std::string configServers = config.read<std::string>("servers", "");
96  std::vector<std::string> listenAddresses =
97  Core::StringUtil::split(configServers, ';');
98  if (listenAddresses.empty()) {
99  PANIC("No server addresses specified to listen on. "
100  "You must set the 'servers' configuration option.");
101  }
102  std::string error = "Server ID has no matching address";
103  for (uint32_t i = 0; i < listenAddresses.size(); ++i) {
104  if (serverId != 0 && serverId != i + 1)
105  continue;
106  RPC::Address address(listenAddresses[i],
107  Protocol::Common::DEFAULT_PORT);
108  error = rpcServer->bind(address);
109  if (error.empty()) {
110  NOTICE("Serving on %s", address.toString().c_str());
111  raft->serverId = i + 1;
112  raft->serverAddress = listenAddresses[i];
113  Core::Debug::processName =
114  Core::StringUtil::format("%lu", raft->serverId);
115  raft->init();
116  break;
117  }
118  }
119  if (!error.empty()) {
120  PANIC("Could not bind to any server address in: %s. "
121  "Last error was: %s",
122  configServers.c_str(),
123  error.c_str());
124  }
125  }
126 
127  if (!stateMachine) {
128  stateMachine.reset(new StateMachine(raft, config));
129  }
130 
131 }
