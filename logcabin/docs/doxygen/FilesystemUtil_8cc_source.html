140 std::vector<std::string>
141 lsHelper(DIR* dir, const std::string& path)
142 {
143  if (dir == NULL) {
144  PANIC("Could not list contents of %s: %s",
145  path.c_str(), strerror(errno));
146  }
147 
148  // If dir was opened with fdopendir and was read from previously, this is
149  // needed to rewind the directory, at least on eglibc v2.13. The unit test
150  // "ls_RewindDir" shows the exact problem.
151  rewinddir(dir);
152 
153  std::vector<std::string> contents;
154  while (true) {
155  struct dirent entry;
156  struct dirent* entryp;
157  if (readdir_r(dir, &entry, &entryp) != 0) {
158  PANIC("readdir(%s) failed: %s",
159  path.c_str(), strerror(errno));
160  }
161  if (entryp == NULL) // no more entries
162  break;
163  const std::string name = entry.d_name;
164  if (name == "." || name == "..")
165  continue;
166  contents.push_back(name);
167  }
168 
169  if (closedir(dir) != 0) {
170  WARNING("closedir(%s) failed: %s",
171  path.c_str(), strerror(errno));
172  }
173 
174  return contents;
175 }

190 openDir(const std::string& path)
191 {
192  assert(!path.empty());
193  int r = mkdir(path.c_str(), 0755);
194  if (r == 0) {
195  FilesystemUtil::syncDir(path + "/..");
196  } else {
197  if (errno != EEXIST) {
198  PANIC("Could not create directory %s: %s",
199  path.c_str(), strerror(errno));
200  }
201  }

203  // but at least on eglibc v2.13, this combination of flags creates a
204  // regular file!

206  if (fd == -1) {
207  PANIC("Could not open %s: %s",
208  path.c_str(), strerror(errno));
209  }

264 remove(const std::string& path)
265 {
266  while (true) {
267  if (::remove(path.c_str()) == 0)
268  return;
269  if (errno == ENOENT) {
270  return;
271  } else if (errno == EEXIST || errno == ENOTEMPTY) {
272  std::vector<std::string> children = ls(path);
273  for (auto it = children.begin(); it != children.end(); ++it)
274  remove(path + "/" + *it);
275  continue;
276  } else {
277  PANIC("Could not remove %s: %s", path.c_str(), strerror(errno));
278  }

360 ssize_t
361 write(int fildes,
362  std::initializer_list<std::pair<const void*, uint64_t>> data)
363 {
364  using Core::Util::downCast;
365  size_t totalBytes = 0;
366  uint64_t iovcnt = data.size();
367  struct iovec iov[iovcnt];
368  uint64_t i = 0;
369  for (auto it = data.begin(); it != data.end(); ++it) {
370  iov[i].iov_base = const_cast<void*>(it->first);
371  iov[i].iov_len = it->second;
372  totalBytes += it->second;
373  ++i;
374  }
375 
376  size_t bytesRemaining = totalBytes;
377  while (true) {
378  ssize_t written = System::writev(fildes, iov, downCast<int>(iovcnt));
379  if (written == -1) {
380  if (errno == EINTR)
381  continue;
382  return -1;
383  }
384  bytesRemaining = downCast<size_t>(bytesRemaining -
385  downCast<size_t>(written));
386  if (bytesRemaining == 0)
387  return downCast<ssize_t>(totalBytes);
388  for (uint64_t i = 0; i < iovcnt; ++i) {
389  if (iov[i].iov_len < static_cast<size_t>(written)) {
390  written -= iov[i].iov_len;
391  iov[i].iov_len = 0;
392  } else if (iov[i].iov_len >= static_cast<size_t>(written)) {
393  iov[i].iov_len = iov[i].iov_len - downCast<size_t>(written);
394  iov[i].iov_base = (static_cast<char*>(iov[i].iov_base) +
395  written);
396  break;
397  }
398  }
399  }
