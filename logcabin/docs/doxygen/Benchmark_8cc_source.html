140  * The main function for a single client thread.
141  * \param id
142  * Unique ID for this thread, counting from 0.
143  * \param options

156 void
157 writeThreadMain(uint64_t id,
158  const OptionParser& options,
159  Tree tree,
160  const std::string& key,
161  const std::string& value,
162  std::atomic<bool>& exit,
163  uint64_t& writesDone)
164 {
165  uint64_t numWrites = options.totalWrites / options.writers;
166  // assign any odd leftover writes in a balanced way
167  if (options.totalWrites - numWrites * options.writers > id)
168  numWrites += 1;
169  for (uint64_t i = 0; i < numWrites; ++i) {
170  if (exit)
171  break;
172  tree.writeEx(key, value);
173  writesDone = i + 1;
174  }
175 }

188 /**
189  * Main function for the timer thread, whose job is to wait until a particular
190  * timeout elapses and then set 'exit' to true.
191  * \param timeout
192  * Seconds to wait before setting exit to true.
193  * \param[in,out] exit
194  * If this is set to true from another thread, the timer thread will exit
195  * soonish. Also, if the timeout elapses, the timer thread will set this
196  * to true and exit.
197  */
198 void
199 timerThreadMain(uint64_t timeout, std::atomic<bool>& exit)
200 {
201  uint64_t start = timeNanos();
202  while (!exit) {
203  usleep(50 * 1000);
204  if ((timeNanos() - start) > timeout * 1000 * 1000 * 1000) {
205  exit = true;
206  }
207  }
208 }

213 main(int argc, char** argv)
214 {
215  OptionParser options(argc, argv);
216  Cluster cluster = Cluster(options.cluster);
217  Tree tree = cluster.getTree();
218 
219  std::string key("/bench");
220  std::string value(options.size, 'v');
221 
222  uint64_t startNanos = timeNanos();
223  std::atomic<bool> exit(false);
224  std::vector<uint64_t> writesDonePerThread(options.writers);
225  uint64_t totalWritesDone = 0;
226  std::vector<std::thread> threads;
227  std::thread timer(timerThreadMain, options.timeout, std::ref(exit));
228  for (uint64_t i = 0; i < options.writers; ++i) {
229  threads.emplace_back(writeThreadMain, i, std::ref(options),
230  tree, std::ref(key), std::ref(value),
231  std::ref(exit),
232  std::ref(writesDonePerThread.at(i)));
233  }
234  for (uint64_t i = 0; i < options.writers; ++i) {
235  threads.at(i).join();
236  totalWritesDone += writesDonePerThread.at(i);
237  }
238  uint64_t endNanos = timeNanos();
239  exit = true;
240  timer.join();
241 
242  tree.removeFile(key);
243  std::cout << "Benchmark took "
244  << static_cast<double>(endNanos - startNanos) / 1e6
245  << " ms to write "
246  << totalWritesDone
247  << " objects"
248  << std::endl;
249  return 0;
