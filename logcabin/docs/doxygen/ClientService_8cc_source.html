32 uint64_t timeNanos()
33 {
34  struct timespec now;
35  int r = clock_gettime(CLOCK_REALTIME, &now);
36  assert(r == 0);
37  return uint64_t(now.tv_sec) * 1000 * 1000 * 1000 + now.tv_nsec;

54 void
55 ClientService::handleRPC(RPC::ServerRPC rpc)
56 {
57  using Protocol::Client::OpCode;
58 
59  // TODO(ongaro): If this is not the current cluster leader, need to
60  // redirect the client.
61 
62  // Call the appropriate RPC handler based on the request's opCode.
63  switch (rpc.getOpCode()) {
64  case OpCode::GET_SUPPORTED_RPC_VERSIONS:
65  getSupportedRPCVersions(std::move(rpc));
66  break;
67  case OpCode::GET_CONFIGURATION:
68  getConfiguration(std::move(rpc));
69  break;
70  case OpCode::SET_CONFIGURATION:
71  setConfiguration(std::move(rpc));
72  break;
73  case OpCode::OPEN_SESSION:
74  openSession(std::move(rpc));
75  break;
76  case OpCode::READ_ONLY_TREE:
77  readOnlyTreeRPC(std::move(rpc));
78  break;
79  case OpCode::READ_WRITE_TREE:
80  readWriteTreeRPC(std::move(rpc));
81  break;
82  default:
83  rpc.rejectInvalidRequest();
84  }

122 std::pair<Result, uint64_t>
123 ClientService::submit(RPC::ServerRPC& rpc,
124  const google::protobuf::Message& command)
125 {
126  // TODO(ongaro): Switch from string to binary format. This is probably
127  // really slow to serialize.
128  std::string cmdStr = Core::ProtoBuf::dumpString(command);
129  std::pair<Result, uint64_t> result = globals.raft->replicate(cmdStr);
130  if (result.first == Result::RETRY || result.first == Result::NOT_LEADER) {
131  Protocol::Client::Error error;
132  error.set_error_code(Protocol::Client::Error::NOT_LEADER);
133  std::string leaderHint = globals.raft->getLeaderHint();
134  if (!leaderHint.empty())
135  error.set_leader_hint(leaderHint);
136  rpc.returnError(error);
137  }
138  return result;
139 }

189 void
190 ClientService::getConfiguration(RPC::ServerRPC rpc)
191 {
192  PRELUDE(GetConfiguration);
193  Protocol::Raft::SimpleConfiguration configuration;
194  uint64_t id;
195  Result result = globals.raft->getConfiguration(configuration, id);
196  if (result == Result::RETRY || result == Result::NOT_LEADER) {
197  Protocol::Client::Error error;
198  error.set_error_code(Protocol::Client::Error::NOT_LEADER);
199  std::string leaderHint = globals.raft->getLeaderHint();
200  if (!leaderHint.empty())
201  error.set_leader_hint(leaderHint);
202  rpc.returnError(error);
203  return;
204  }
205  response.set_id(id);
206  for (auto it = configuration.servers().begin();
207  it != configuration.servers().end();
208  ++it) {
209  Protocol::Client::Server* server = response.add_servers();
210  server->set_server_id(it->server_id());
211  server->set_address(it->address());
212  }
213  rpc.reply(response);
214 }

216 void
217 ClientService::setConfiguration(RPC::ServerRPC rpc)
218 {
219  PRELUDE(SetConfiguration);
220  Protocol::Raft::SimpleConfiguration newConfiguration;
221  for (auto it = request.new_servers().begin();
222  it != request.new_servers().end();
223  ++it) {
224  Protocol::Raft::Server* s = newConfiguration.add_servers();
225  s->set_server_id(it->server_id());
226  s->set_address(it->address());
227  }
228  Result result = globals.raft->setConfiguration(
229  request.old_id(),
230  newConfiguration);
231  if (result == Result::SUCCESS) {
232  response.mutable_ok();
233  } else if (result == Result::RETRY || result == Result::NOT_LEADER) {
234  Protocol::Client::Error error;
235  error.set_error_code(Protocol::Client::Error::NOT_LEADER);
236  std::string leaderHint = globals.raft->getLeaderHint();
237  if (!leaderHint.empty())
238  error.set_leader_hint(leaderHint);
239  rpc.returnError(error);
240  return;
241  } else if (result == Result::FAIL) {
242  // TODO(ongaro): can't distinguish changed from bad
243  response.mutable_configuration_changed();
244  }
245  rpc.reply(response);
246 }

258 void
259 ClientService::readWriteTreeRPC(RPC::ServerRPC rpc)
260 {
261  PRELUDE(ReadWriteTree);
262  Command command;
263  command.set_nanoseconds_since_epoch(timeNanos());
264  *command.mutable_tree() = request;
265  std::pair<Result, uint64_t> result = submit(rpc, command);
266  if (result.first != Result::SUCCESS)
267  return;
268  CommandResponse commandResponse;
269  if (!getResponse(rpc, result.second, request.exactly_once(),
270  commandResponse)) {
271  return;
272  }
273  rpc.reply(commandResponse.tree());
274 }
