37 // Internal scratch state used by random_r 128 is the same size as
38 // initstate() uses for regular random(), see manpages for details.
39 // statebuf is malloc'ed and this memory is leaked, it could be a __thread
40 // buffer, but after running into linker issues with large thread local
41 // storage buffers, we thought better.
42 enum { STATE_BYTES = 128 };
43 __thread char* statebuf;
44 // random_r's state, must be handed to each call, and seems to refer to
45 // statebuf in some undocumented way.
46 __thread random_data randbuf;

48 /**
49  * Initialize thread-local state for random number generator.
50  * Must be called before any random numbers are generated.
51  */
52 void
53 initRandom()
54 {
55  if (statebuf == NULL) {
56  int fd = open("/dev/urandom", O_RDONLY);
57  if (fd < 0)
58  PANIC("Couldn't open /dev/urandom: %s", strerror(errno));
59  unsigned int seed;
60  ssize_t bytesRead = read(fd, &seed, sizeof(seed));
61  close(fd);
62  if (bytesRead != sizeof(seed))
63  PANIC("Couldn't read full seed from /dev/urandom");
64  statebuf = static_cast<char*>(malloc(STATE_BYTES));
65  initstate_r(seed, statebuf, STATE_BYTES, &randbuf);
66  }
67 
68  static_assert(RAND_MAX >= (1 << 31), "RAND_MAX too small");
69 }

71 /**
72  * Fill a variable of type T with some random bytes.
73  */
74 template<typename T>
75 T
76 getRandomBytes()
77 {
78  T buf;
79  size_t offset = 0;
80  while (offset < sizeof(buf)) {
81  uint64_t r = random64();
82  size_t copy = std::min(sizeof(r), sizeof(buf) - offset);
83  memcpy(reinterpret_cast<char*>(&buf) + offset, &r, copy);
84  offset += copy;
85  }
86  return buf;
87 }

117 uint64_t
118 random64()
119 {
120  // Each call to random returns 31 bits of randomness,
121  // so we need three to get 64 bits of randomness.
122  initRandom();
123  int32_t lo, mid, hi;
124  random_r(&randbuf, &lo);
125  random_r(&randbuf, &mid);
126  random_r(&randbuf, &hi);
127  uint64_t r = (((uint64_t(hi) & 0x7FFFFFFF) << 33) | // NOLINT
128  ((uint64_t(mid) & 0x7FFFFFFF) << 2) | // NOLINT
129  (uint64_t(lo) & 0x00000003)); // NOLINT
130  return r;
131 }
