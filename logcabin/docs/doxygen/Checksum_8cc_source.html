60  /**
61  * Calculate the checksum for some data.
62  * \param data
63  * An list of (pointer, length) pairs describing what to checksum.
64  * \param[out] result
65  * The result of the hash function will be placed here.
66  * This will be a null-terminated, printable C-string.
67  * \return
68  * The number of valid characters in 'output', including the null
69  * terminator. This is guaranteed to be greater than 1.
70  */
71  uint32_t
72  writeChecksum(std::initializer_list<std::pair<const void*, uint64_t>> data,
73  char result[MAX_LENGTH]) {
74  // copy name and : to result
75  memcpy(result, name.c_str(), name.length());
76  result = result + name.length();
77  *result = ':';
78  ++result;
79 
80  // calculate binary digest
81  uint32_t binarySize = downCast<uint32_t>(hashFn->DigestSize());
82  uint8_t binary[binarySize];
83 
84  for (auto it = data.begin(); it != data.end(); ++it) {
85  hashFn->Update(static_cast<const uint8_t*>(it->first),
86  it->second);
87  }
88  hashFn->Final(binary);
89 
90  // add hex digest to result
91  const char* hexArray = "0123456789abcdef";
92  for (uint32_t i = 0; i < binarySize; ++i) {
93  *result = hexArray[binary[i] >> 4];
94  ++result;
95  *result = hexArray[binary[i] & 15];
96  ++result;
97  }
98 
99  // add null terminator to result and return total length
100  *result = '\0';
101  return outputSize;
102  }
103 
104  protected:
105  // Called by derived class's constructor to initialize this class
106  void setHashFn(CryptoPP::HashTransformation& hashFn) {
107  const std::string name = hashFn.AlgorithmName();
108  size_t outputSize = name.length() + 1 + hashFn.DigestSize() * 2 + 1;
109  assert(outputSize <= MAX_LENGTH);
110 
111  this->hashFn = &hashFn;
112  this->name = name;
113  this->outputSize = downCast<uint32_t>(outputSize);
114  }
115 
116  private:

118  std::string name;
119  uint32_t outputSize;

288 std::string
289 verify(const char* checksum,
290  std::initializer_list<std::pair<const void*, uint64_t>> data)
291 {
292  if (!Core::StringUtil::isPrintable(checksum))
293  return "The given checksum value is corrupt and not printable.";
294 
295  Algorithm* algo;
296  { // find algo
297  char algorithmName[MAX_LENGTH];
298  char* colon = strchr(const_cast<char*>(checksum), ':');
299  if (colon == NULL)
300  return format("Missing colon in checksum: %s", checksum);
301 
302  // nameLength does not include the null terminator
303  uint32_t nameLength = downCast<uint32_t>(colon - checksum);
304  memcpy(algorithmName, checksum, nameLength);
305  algorithmName[nameLength] = '\0';
306 
307  algo = Algorithms::find(algorithmName);
308  if (algo == NULL)
309  return format("No such checksum algorithm: %s",
310  algorithmName);
311  }
312 
313  // compare calculated checksum with the one given
314  char calculated[MAX_LENGTH];
315  algo->writeChecksum(data, calculated);
316  if (strcmp(calculated, checksum) != 0) {
317  return format("Checksum doesn't match: expected %s "
318  "but calculated %s", checksum, calculated);
319  }
320 
321  return std::string();
322 }
