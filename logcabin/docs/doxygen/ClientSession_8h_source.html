41  * A ClientSession is used to initiate OpaqueClientRPCs. It encapsulates a
42  * connection to a server. Sessions can be relatively expensive to create, so
43  * clients should keep them around.
44  */

55  /**
56  * Return a new ClientSession object. This object is managed by a
57  * std::shared_ptr to ensure that it remains alive while there are
58  * outstanding RPCs.
59  *
60  * This should only be used from worker threads, as it invokes possibly
61  * long-running syscalls.
62  *
63  * \param eventLoop
64  * Event::Loop that will be used to find out when the underlying
65  * socket may be read from or written to without blocking.
66  * \param address
67  * The RPC server address on which to connect.
68  * \param maxMessageLength
69  * The maximum number of bytes to allow per request/response. This

137  /**
138  * This contains an expected response for a OpaqueClientRPC object.
139  * This is created when the OpaqueClientRPC is created; it is deleted when
140  * the OpaqueClientRPC object is either canceled or updated with a
141  * response/error.
142  */
143  struct Response {
144  /**
145  * Constructor.
146  */
147  Response();
148  /**
149  * Current state of the RPC.
150  */
151  enum {
152  /**
153  * Waiting for a reply from the server.
154  */
155  WAITING,
156  /**
157  * Received a reply (find it in #reply).
158  */
159  HAS_REPLY,
160  /**
161  * The RPC has been canceled by another thread.
162  */
163  CANCELED,
164  } status;
165  /**
166  * The contents of the response. This is valid when
167  * #status is HAS_REPLY.
168  */
169  Buffer reply;
170  /**
171  * If true, a thread is blocked waiting on #ready,
172  * and this object may not be deleted.
173  */
174  bool hasWaiter;
175  /**
176  * OpaqueClientRPC objects wait on this condition variable inside of
177  * wait(). It is notified when a new response arrives, the session
178  * is disconnected, or the RPC is canceled.
179  */
180  Core::ConditionVariable ready;
181  };
