238 bool
239 Peer::callRPC(Protocol::Raft::OpCode opCode,
240  const google::protobuf::Message& request,
241  google::protobuf::Message& response,
242  std::unique_lock<Mutex>& lockGuard)
243 {
244  typedef RPC::ClientRPC::Status RPCStatus;
245  rpc = RPC::ClientRPC(getSession(lockGuard),
246  Protocol::Common::ServiceId::RAFT_SERVICE,
247  /* serviceSpecificErrorVersion = */ 0,
248  opCode,
249  request);
250  // release lock for concurrency
251  Core::MutexUnlock<Mutex> unlockGuard(lockGuard);
252  switch (rpc.waitForReply(&response, NULL)) {
253  case RPCStatus::OK:
254  if (rpcFailuresSinceLastWarning > 0) {
255  WARNING("RPC to server succeeded after %lu failures",
256  rpcFailuresSinceLastWarning);
257  rpcFailuresSinceLastWarning = 0;
258  }
259  return true;
260  case RPCStatus::SERVICE_SPECIFIC_ERROR:
261  PANIC("unexpected service-specific error");
262  default:
263  ++rpcFailuresSinceLastWarning;
264  if (rpcFailuresSinceLastWarning == 1) {
265  WARNING("RPC to server failed: %s",
266  rpc.getErrorMessage().c_str());
267  } else if (rpcFailuresSinceLastWarning % 100 == 0) {
268  WARNING("Last %lu RPCs to server failed. This failure: %s",
269  rpcFailuresSinceLastWarning,
270  rpc.getErrorMessage().c_str());
271  }
272  return false;
273  }
274 }

502 void
503 Configuration::setConfiguration(
504  uint64_t newId,
505  const Protocol::Raft::Configuration& newDescription)
506 {
507  NOTICE("Activating configuration %lu:\n%s", newId,
508  Core::ProtoBuf::dumpString(newDescription).c_str());
509 
510  if (newDescription.next_configuration().servers().size() == 0)
511  state = State::STABLE;
512  else
513  state = State::TRANSITIONAL;
514  id = newId;
515  description = newDescription;
516  oldServers.servers.clear();
517  newServers.servers.clear();
518 
519  // Build up the list of old servers
520  for (auto confIt = description.prev_configuration().servers().begin();
521  confIt != description.prev_configuration().servers().end();
522  ++confIt) {
523  std::shared_ptr<Server> server = getServer(confIt->server_id());
524  server->address = confIt->address();
525  oldServers.servers.push_back(server);
526  }
527 
528  // Build up the list of new servers
529  for (auto confIt = description.next_configuration().servers().begin();
530  confIt != description.next_configuration().servers().end();
531  ++confIt) {
532  std::shared_ptr<Server> server = getServer(confIt->server_id());
533  server->address = confIt->address();
534  newServers.servers.push_back(server);
535  }
536 
537  // Servers not in the current configuration need to be told to exit
538  setGCFlag(*localServer);
539  oldServers.forEach(setGCFlag);
540  newServers.forEach(setGCFlag);
541  auto it = knownServers.begin();
542  while (it != knownServers.end()) {
543  std::shared_ptr<Server> server = it->second;
544  if (!server->gcFlag) {
545  server->exit();
546  it = knownServers.erase(it);
547  } else {
548  server->gcFlag = false; // clear flag for next time
549  ++it;
550  }
551  }
552 }

920 RaftConsensus::getNextEntry(uint64_t lastEntryId) const
921 {

923  uint64_t nextEntryId = lastEntryId + 1;
924  while (true) {
925  if (exiting)
926  throw ThreadInterruptedException();

945  // snapshot should have already been read, so const_cast
946  // should be ok (though ugly).
947  const_cast<RaftConsensus*>(this)->readSnapshot();

957  entry.data = logEntry.data();
958  } else {
959  entry.type = Consensus::Entry::SKIP;
960  }
961  }
962  return entry;
963  }
964  stateChanged.wait(lockGuard);
965  }

983 void
984 RaftConsensus::handleAppendEntries(
985  const Protocol::Raft::AppendEntries::Request& request,
986  Protocol::Raft::AppendEntries::Response& response)
987 {
988  std::unique_lock<Mutex> lockGuard(mutex);
989  assert(!exiting);
990 
991  // Set response to a rejection. We'll overwrite these later if we end up
992  // accepting the request.
993  response.set_term(currentTerm);
994  response.set_success(false);
995 
996  // If the caller's term is stale, just return our term to it.
997  if (request.term() < currentTerm) {
998  VERBOSE("Caller(%lu) is stale. Our term is %lu, theirs is %lu",
999  request.server_id(), currentTerm, request.term());
1000  return; // response was set to a rejection above
1001  }
1002  if (request.term() > currentTerm) {
1003  VERBOSE("Caller(%lu) has newer term, updating. "
1004  "Ours was %lu, theirs is %lu",
1005  request.server_id(), currentTerm, request.term());
1006  // We're about to bump our term in the stepDown below: update
1007  // 'response' accordingly.
1008  response.set_term(request.term());
1009  }
1010  // This request is a sign of life from the current leader. Update our term
1011  // and convert to follower if necessary; reset the election timer.
1012  stepDown(request.term());
1013  setElectionTimer();
1014  withholdVotesUntil = Clock::now() +
1015  std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
1016 
1017  // Record the leader ID as a hint for clients.
1018  if (leaderId == 0) {
1019  leaderId = request.server_id();
1020  NOTICE("All hail leader %lu for term %lu", leaderId, currentTerm);
1021  } else {
1022  assert(leaderId == request.server_id());
1023  }
1024 
1025  // For an entry to fit into our log, it must not leave a gap.
1026  if (request.prev_log_index() > log->getLastLogIndex()) {
1027  VERBOSE("Rejecting AppendEntries RPC: would leave gap");
1028  return; // response was set to a rejection above
1029  }
1030  // It must also agree with the previous entry in the log (and, inductively
1031  // all prior entries).
1032  // Always match on index 0, and always match on any discarded indexes:
1033  // since we know those were committed, the leader must agree with them.
1034  // We could truncate the log here, but there's no real advantage to doing
1035  // that.
1036  if (request.prev_log_index() >= log->getLogStartIndex() &&
1037  log->getEntry(request.prev_log_index()).term() !=
1038  request.prev_log_term()) {
1039  VERBOSE("Rejecting AppendEntries RPC: terms don't agree");
1040  return; // response was set to a rejection above
1041  }
1042 
1043  // If we got this far, we're accepting the request.
1044  response.set_success(true);
1045 
1046  // This needs to be able to handle duplicated RPC requests. We compare the
1047  // entries' terms to know if we need to do the operation; otherwise,
1048  // reapplying requests can result in data loss.
1049  //
1050  // The first problem this solves is that an old AppendEntries request may be
1051  // duplicated and received after a newer request, which could cause
1052  // undesirable data loss. For example, suppose the leader appends entry 4
1053  // and then entry 5, but the follower receives 4, then 5, then 4 again.
1054  // Without this extra guard, the follower would truncate 5 out of its
1055  // log.
1056  //
1057  // The second problem is more subtle: if the same request is duplicated but
1058  // the leader processes an earlier response, it will assume the
1059  // acknowledged data is safe. However, there is a window of vulnerability
1060  // on the follower's disk between the truncate and append operations (which
1061  // are not done atomically) when the follower processes the later request.
1062  uint64_t entryId = request.prev_log_index();
1063  for (auto it = request.entries().begin();
1064  it != request.entries().end();
1065  ++it) {
1066  ++entryId;
1067  const Protocol::Raft::Entry& entry = *it;
1068  if (entryId < log->getLogStartIndex()) {
1069  // We already snapshotted and discarded this index, so presumably
1070  // we've received a committed entry we once already had.
1071  continue;
1072  }
1073  if (log->getLastLogIndex() >= entryId) {
1074  if (log->getEntry(entryId).term() == entry.term())
1075  continue;
1076  // should never truncate committed entries:
1077  assert(commitIndex < entryId);
1078  NOTICE("Truncating %lu entries after %lu from the log",
1079  log->getLastLogIndex() - entryId + 1,
1080  entryId - 1);
1081  log->truncateSuffix(entryId - 1);
1082  configurationManager->truncateSuffix(entryId - 1);
1083  }
1084 
1085  // Append this and all following entries.
1086  std::vector<const Protocol::Raft::Entry*> entries;
1087  while (it != request.entries().end()) {
1088  entries.push_back(&*it);
1089  ++it;
1090  }
1091  append(entries);
1092  break;
1093  }
1094 
1095  // Set our committed ID from the request's. In rare cases, this would make
1096  // our committed ID decrease. For example, this could happen with a new
1097  // leader who has not yet replicated one of its own entries. While that'd
1098  // be perfectly safe, guarding against it with an if statement lets us
1099  // make stronger assertions.
1100  if (commitIndex < request.commit_index()) {
1101  commitIndex = request.commit_index();
1102  assert(commitIndex <= log->getLastLogIndex());
1103  stateChanged.notify_all();
1104  VERBOSE("New commitIndex: %lu", commitIndex);
1105  }
1106 }

1253 RaftConsensus::ClientResult
1254 RaftConsensus::setConfiguration(
1255  uint64_t oldId,
1256  const Protocol::Raft::SimpleConfiguration& nextConfiguration)
1257 {
1258  std::unique_lock<Mutex> lockGuard(mutex);
1259 
1260  if (state != State::LEADER)
1261  return ClientResult::NOT_LEADER;
1262  if (configuration->id != oldId ||
1263  configuration->state != Configuration::State::STABLE) {
1264  // configurations has changed in the meantime
1265  return ClientResult::FAIL;
1266  }
1267 
1268  uint64_t term = currentTerm;
1269  configuration->setStagingServers(nextConfiguration);
1270  stateChanged.notify_all();
1271 
1272  // Wait for new servers to be caught up. This will abort if not every
1273  // server makes progress in a ELECTION_TIMEOUT_MS period.
1274  ++currentEpoch;
1275  uint64_t epoch = currentEpoch;
1276  TimePoint checkProgressAt =
1277  Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
1278  while (true) {
1279  if (exiting || term != currentTerm)
1280  return ClientResult::NOT_LEADER;
1281  if (configuration->stagingAll(&Server::isCaughtUp))
1282  break;
1283  if (Clock::now() >= checkProgressAt) {
1284  if (configuration->stagingMin(&Server::getLastAckEpoch) < epoch) {
1285  configuration->resetStagingServers();
1286  stateChanged.notify_all();
1287  // TODO(ongaro): probably need to return a different type of
1288  // message: confuses oldId mismatch from new server down
1289  return ClientResult::FAIL;
1290  } else {
1291  ++currentEpoch;
1292  epoch = currentEpoch;
1293  checkProgressAt =
1294  (Clock::now() +
1295  std::chrono::milliseconds(ELECTION_TIMEOUT_MS));
1296  }
1297  }
1298  stateChanged.wait_until(lockGuard, checkProgressAt);
1299  }
1300 
1301  // Write and commit transitional configuration
1302  Protocol::Raft::Configuration newConfiguration;
1303  *newConfiguration.mutable_prev_configuration() =
1304  configuration->description.prev_configuration();
1305  *newConfiguration.mutable_next_configuration() = nextConfiguration;
1306  Log::Entry entry;
1307  entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
1308  *entry.mutable_configuration() = newConfiguration;
1309  std::pair<ClientResult, uint64_t> result =
1310  replicateEntry(entry, lockGuard);
1311  if (result.first != ClientResult::SUCCESS)
1312  return result.first;
1313  uint64_t transitionalId = result.second;
1314 
1315  // Wait until the configuration that removes the old servers has been
1316  // committed. This is the first configuration with ID greater than
1317  // transitionalId.
1318  while (true) {
1319  // Check this first: if the new configuration excludes us so we've
1320  // stepped down upon committing it, we still want to return success.
1321  if (configuration->id > transitionalId &&
1322  commitIndex >= configuration->id) {
1323  return ClientResult::SUCCESS;
1324  }
1325  if (exiting || term != currentTerm)
1326  return ClientResult::NOT_LEADER;
1327  stateChanged.wait(lockGuard);
1328  }
1329 }

1331 std::unique_ptr<Storage::SnapshotFile::Writer>
1332 RaftConsensus::beginSnapshot(uint64_t lastIncludedIndex)
1333 {
1334  std::unique_lock<Mutex> lockGuard(mutex);
1335 
1336  NOTICE("Creating new snapshot through log index %lu (inclusive)",
1337  lastIncludedIndex);
1338  std::unique_ptr<Storage::SnapshotFile::Writer> writer(
1339  new Storage::SnapshotFile::Writer(storageDirectory));
1340 
1341  // Only committed entries may be snapshotted.
1342  // (This check relies on commitIndex monotonically increasing.)
1343  if (lastIncludedIndex > commitIndex) {
1344  PANIC("Attempted to snapshot uncommitted entries (%lu requested but "
1345  "%lu is last committed entry)", lastIncludedIndex, commitIndex);
1346  }
1347 
1348  // set header fields
1349  SnapshotMetadata::Header header;
1350  header.set_last_included_index(lastIncludedIndex);
1351  // Set last_included_term:
1352  if (lastIncludedIndex >= log->getLogStartIndex() &&
1353  lastIncludedIndex <= log->getLastLogIndex()) {
1354  header.set_last_included_term(log->getEntry(lastIncludedIndex).term());
1355  } else if (lastIncludedIndex == 0) {
1356  WARNING("Taking a snapshot covering no log entries");
1357  header.set_last_included_term(0);
1358  } else if (lastIncludedIndex == lastSnapshotIndex) {
1359  WARNING("Taking a snapshot where we already have one, covering "
1360  "entries 1 through %lu (inclusive)", lastIncludedIndex);
1361  header.set_last_included_term(lastSnapshotTerm);
1362  } else {
1363  WARNING("We've already discarded the entries that the state machine "
1364  "wants to snapshot. This can happen in rare cases if the "
1365  "leader already sent us a newer snapshot. We'll go ahead and "
1366  "compute the snapshot, but it'll be discarded later in "
1367  "snapshotDone(). Setting the last included term in the "
1368  "snapshot header to 0 (a bogus value).");
1369  // If this turns out to be common, we should return NULL instead and
1370  // change the state machines to deal with that.
1371  header.set_last_included_term(0);
1372  }
1373  // Copy the configuration as of lastIncludedIndex to the header.
1374  std::pair<uint64_t, Protocol::Raft::Configuration> c =
1375  configurationManager->getLatestConfigurationAsOf(lastIncludedIndex);
1376  if (c.first == 0) {
1377  WARNING("Taking snapshot with no configuration. "
1378  "This should have been the first thing in the log.");
1379  } else {
1380  header.set_configuration_index(c.first);
1381  *header.mutable_configuration() = c.second;
1382  }
1383 
1384  // write header to file
1385  google::protobuf::io::CodedOutputStream& stream = writer->getStream();
1386  int size = header.ByteSize();
1387  stream.WriteLittleEndian32(size);
1388  header.SerializeWithCachedSizes(&stream);
1389 
1390  return writer;
1391 }

1393 void
1394 RaftConsensus::snapshotDone(
1395  uint64_t lastIncludedIndex,
1396  std::unique_ptr<Storage::SnapshotFile::Writer> writer)
1397 {
1398  std::unique_lock<Mutex> lockGuard(mutex);
1399  if (lastIncludedIndex <= lastSnapshotIndex) {
1400  NOTICE("Discarding snapshot through %lu since we already have one "
1401  "(presumably from another server) through %lu",
1402  lastIncludedIndex, lastSnapshotIndex);
1403  writer->discard();
1404  return;
1405  }
1406 
1407  // log->getEntry(lastIncludedIndex) is safe:
1408  // If the log prefix for this snapshot was truncated, that means we have a
1409  // newer snapshot (handled above).
1410  assert(lastIncludedIndex >= log->getLogStartIndex());
1411  // We never truncate committed entries from the end of our log, and
1412  // beginSnapshot() made sure that lastIncludedIndex covers only committed
1413  // entries.
1414  assert(lastIncludedIndex <= log->getLastLogIndex());
1415 
1416  lastSnapshotBytes = writer->save();
1417  lastSnapshotIndex = lastIncludedIndex;
1418  lastSnapshotTerm = log->getEntry(lastIncludedIndex).term();
1419

1421  // carry it around after writing the header.

1572 void
1573 RaftConsensus::stepDownThreadMain()
1574 {
1575  std::unique_lock<Mutex> lockGuard(mutex);
1576  Core::ThreadId::setName("stepDown");
1577  while (true) {
1578  // Wait until this server is the leader and is not the only server in
1579  // the cluster.
1580  while (true) {
1581  if (exiting)
1582  return;
1583  if (state == State::LEADER) {
1584  // If this local server forms a quorum (it is the only server
1585  // in the configuration), we need to sleep. Without this guard,
1586  // this method would not relinquish the CPU.
1587  ++currentEpoch;
1588  if (configuration->quorumMin(&Server::getLastAckEpoch) <
1589  currentEpoch) {
1590  break;
1591  }
1592  }
1593  stateChanged.wait(lockGuard);
1594  }
1595  // Now, if an election timeout goes by without confirming leadership,
1596  // step down. The election timeout is a reasonable amount of time,
1597  // since it's about when other servers will start elections and bump
1598  // the term.
1599  TimePoint stepDownAt =
1600  Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
1601  uint64_t term = currentTerm;
1602  uint64_t epoch = currentEpoch; // currentEpoch was incremented above
1603  while (true) {
1604  if (exiting)
1605  return;
1606  if (currentTerm > term)
1607  break;
1608  if (configuration->quorumMin(&Server::getLastAckEpoch) >= epoch)
1609  break;
1610  if (Clock::now() >= stepDownAt) {
1611  NOTICE("No broadcast for a timeout, stepping down");
1612  stepDown(currentTerm + 1);
1613  break;
1614  }
1615  stateChanged.wait_until(lockGuard, stepDownAt);
1616  }
1617  }
1618 }

1622 void
1623 RaftConsensus::advanceCommittedId()
1624 {
1625  if (state != State::LEADER) {
1626  // getLastAgreeIndex is undefined unless we're leader
1627  WARNING("advanceCommittedId called as %s",
1628  Core::StringUtil::toString(state).c_str());
1629  return;
1630  }
1631 
1632  // calculate the largest entry ID stored on a quorum of servers
1633  uint64_t newCommittedId =
1634  configuration->quorumMin(&Server::getLastAgreeIndex);
1635  if (commitIndex >= newCommittedId)
1636  return;
1637  // If we have discarded the entry, it's because we already knew it was
1638  // committed.
1639  assert(newCommittedId >= log->getLogStartIndex());
1640  // At least one of these entries must also be from the current term to
1641  // guarantee that no server without them can be elected.
1642  if (log->getEntry(newCommittedId).term() != currentTerm)
1643  return;
1644  commitIndex = newCommittedId;
1645  VERBOSE("New commitIndex: %lu", commitIndex);
1646  assert(commitIndex <= log->getLastLogIndex());
1647  stateChanged.notify_all();
1648 
1649  if (state == State::LEADER && commitIndex >= configuration->id) {
1650  // Upon committing a configuration that excludes itself, the leader
1651  // steps down.
1652  if (!configuration->hasVote(configuration->localServer)) {
1653  stepDown(currentTerm + 1);
1654  return;
1655  }
1656 
1657  // Upon committing a reconfiguration (Cold,new) entry, the leader
1658  // creates the next configuration (Cnew) entry.
1659  if (configuration->state == Configuration::State::TRANSITIONAL) {
1660  Log::Entry entry;
1661  entry.set_term(currentTerm);
1662  entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
1663  *entry.mutable_configuration()->mutable_prev_configuration() =
1664  configuration->description.next_configuration();
1665  append({&entry});
1666  return;
1667  }
1668  }

1694 void
1695 RaftConsensus::appendEntries(std::unique_lock<Mutex>& lockGuard,
1696  Peer& peer)
1697 {
1698  uint64_t lastLogIndex = log->getLastLogIndex();
1699  uint64_t prevLogIndex = peer.nextIndex - 1;
1700  assert(prevLogIndex <= lastLogIndex);
1701 
1702  // Don't have needed entry: send a snapshot instead.
1703  if (peer.nextIndex < log->getLogStartIndex()) {
1704  appendSnapshotChunk(lockGuard, peer);
1705  return;
1706  }
1707 
1708  // Find prevLogTerm or fall back to sending a snapshot.
1709  uint64_t prevLogTerm;
1710  if (prevLogIndex >= log->getLogStartIndex()) {
1711  prevLogTerm = log->getEntry(prevLogIndex).term();
1712  } else if (prevLogIndex == 0) {
1713  prevLogTerm = 0;
1714  } else if (prevLogIndex == lastSnapshotIndex) {
1715  prevLogTerm = lastSnapshotTerm;
1716  } else {
1717  // Don't have needed entry for prevLogTerm: send snapshot instead.
1718  appendSnapshotChunk(lockGuard, peer);
1719  return;
1720  }
1721 
1722  // Build up request
1723  Protocol::Raft::AppendEntries::Request request;
1724  request.set_server_id(serverId);
1725  request.set_recipient_id(peer.serverId);
1726  request.set_term(currentTerm);
1727  request.set_prev_log_term(prevLogTerm);
1728  request.set_prev_log_index(prevLogIndex);
1729 
1730  // Add as many as entries as will fit comfortably in the request. It's
1731  // easiest to add one entry at a time until the RPC gets too big, then back
1732  // the last one out.
1733  uint64_t numEntries = 0;
1734  if (!peer.forceHeartbeat) {
1735  for (uint64_t entryId = peer.nextIndex;
1736  entryId <= lastLogIndex;
1737  ++entryId) {
1738  const Log::Entry& entry = log->getEntry(entryId);
1739  *request.add_entries() = entry;
1740  uint64_t requestSize =
1741  Core::Util::downCast<uint64_t>(request.ByteSize());
1742  if (requestSize < SOFT_RPC_SIZE_LIMIT || numEntries == 0) {
1743  // this entry fits, send it
1744  VERBOSE("sending entry <id=%lu,term=%lu>",
1745  entryId, entry.term());
1746  ++numEntries;
1747  } else {
1748  // this entry doesn't fit, discard it
1749  request.mutable_entries()->RemoveLast();
1750  }
1751  }
1752  }
1753  request.set_commit_index(std::min(commitIndex, prevLogIndex + numEntries));
1754 
1755  // Execute RPC
1756  Protocol::Raft::AppendEntries::Response response;
1757  TimePoint start = Clock::now();
1758  uint64_t epoch = currentEpoch;
1759  bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_ENTRIES,
1760  request, response,
1761  lockGuard);
1762  if (!ok) {
1763  peer.backoffUntil = start +
1764  std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
1765  return;
1766  }
1767 
1768  // Process response
1769 
1770  if (currentTerm != request.term() || peer.exiting) {
1771  // we don't care about result of RPC
1772  return;
1773  }
1774  // Since we were leader in this term before, we must still be leader in
1775  // this term.
1776  assert(state == State::LEADER);
1777  if (response.term() > currentTerm) {
1778  stepDown(response.term());
1779  } else {
1780  assert(response.term() == currentTerm);
1781  peer.lastAckEpoch = epoch;
1782  stateChanged.notify_all();
1783  peer.nextHeartbeatTime = start +
1784  std::chrono::milliseconds(HEARTBEAT_PERIOD_MS);
1785  if (response.success()) {
1786  if (peer.lastAgreeIndex > prevLogIndex + numEntries) {
1787  // Revisit this warning if we pipeline AppendEntries RPCs for
1788  // performance.
1789  WARNING("lastAgreeIndex should monotonically increase within a "
1790  "term, since servers don't forget entries. But it "
1791  "didn't.");
1792  } else {
1793  peer.lastAgreeIndex = prevLogIndex + numEntries;
1794  advanceCommittedId();
1795  }
1796  peer.nextIndex = peer.lastAgreeIndex + 1;
1797  peer.forceHeartbeat = false;
1798 
1799  if (!peer.isCaughtUp_ &&
1800  peer.thisCatchUpIterationGoalId <= peer.lastAgreeIndex) {
1801  Clock::duration duration =
1802  Clock::now() - peer.thisCatchUpIterationStart;
1803  uint64_t thisCatchUpIterationMs =
1804  uint64_t(std::chrono::duration_cast<
1805  std::chrono::milliseconds>(duration).count());
1806  if (uint64_t(labs(int64_t(peer.lastCatchUpIterationMs -
1807  thisCatchUpIterationMs))) <
1808  ELECTION_TIMEOUT_MS) {
1809  peer.isCaughtUp_ = true;
1810  stateChanged.notify_all();
1811  } else {
1812  peer.lastCatchUpIterationMs = thisCatchUpIterationMs;
1813  peer.thisCatchUpIterationStart = Clock::now();
1814  peer.thisCatchUpIterationGoalId = log->getLastLogIndex();
1815  }
1816  }
1817  } else {
1818  if (peer.nextIndex > 1)
1819  --peer.nextIndex;
1820  }
1821  }
1822 }

1824 void
1825 RaftConsensus::appendSnapshotChunk(std::unique_lock<Mutex>& lockGuard,
1826  Peer& peer)
1827 {
1828  // Build up request
1829  Protocol::Raft::AppendSnapshotChunk::Request request;
1830  request.set_server_id(serverId);
1831  request.set_recipient_id(peer.serverId);
1832  request.set_term(currentTerm);
1833 
1834  // Open the latest snapshot if we haven't already. Stash a copy of the
1835  // lastSnapshotIndex that goes along with the file, since it's possible
1836  // that this will change while we're transferring chunks).
1837  if (!peer.snapshotFile) {
1838  namespace FilesystemUtil = Storage::FilesystemUtil;
1839  peer.snapshotFile.reset(new FilesystemUtil::FileContents(
1840  FilesystemUtil::openFile(storageDirectory, "snapshot", O_RDONLY)));
1841  peer.snapshotFileOffset = 0;
1842  peer.lastSnapshotIndex = lastSnapshotIndex;
1843  }
1844  request.set_last_snapshot_index(peer.lastSnapshotIndex);
1845  request.set_byte_offset(peer.snapshotFileOffset);
1846  // The amount of data we can send is bounded by the remaining bytes in the
1847  // file and the maximum length for RPCs.
1848  uint64_t numDataBytes = std::min(
1849  peer.snapshotFile->getFileLength() - peer.snapshotFileOffset,
1850  SOFT_RPC_SIZE_LIMIT);
1851  request.set_data(peer.snapshotFile->get<char>(peer.snapshotFileOffset,
1852  numDataBytes),
1853  numDataBytes);
1854  request.set_done(peer.snapshotFileOffset + numDataBytes ==
1855  peer.snapshotFile->getFileLength());
1856 
1857  // Execute RPC
1858  Protocol::Raft::AppendSnapshotChunk::Response response;
1859  TimePoint start = Clock::now();
1860  uint64_t epoch = currentEpoch;
1861  bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_SNAPSHOT_CHUNK,
1862  request, response,
1863  lockGuard);
1864  if (!ok) {
1865  peer.backoffUntil = start +
1866  std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
1867  return;
1868  }
1869 
1870  // Process response
1871 
1872  if (currentTerm != request.term() || peer.exiting) {
1873  // we don't care about result of RPC
1874  return;
1875  }
1876  // Since we were leader in this term before, we must still be leader in
1877  // this term.
1878  assert(state == State::LEADER);
1879  if (response.term() > currentTerm) {
1880  stepDown(response.term());
1881  } else {
1882  assert(response.term() == currentTerm);
1883  peer.lastAckEpoch = epoch;
1884  stateChanged.notify_all();
1885  peer.nextHeartbeatTime = start +
1886  std::chrono::milliseconds(HEARTBEAT_PERIOD_MS);
1887  peer.snapshotFileOffset += numDataBytes;
1888  if (request.done()) {
1889  peer.lastAgreeIndex = peer.lastSnapshotIndex;
1890  peer.nextIndex = peer.lastSnapshotIndex + 1;
1891  // These entries are already committed if they're in a snapshot, so
1892  // the commitIndex shouldn't advance, but let's just follow the
1893  // simple rule that bumping lastAgreeIndex should always be
1894  // followed by a call to advanceCommittedId():
1895  advanceCommittedId();
1896  peer.snapshotFile.reset();
1897  peer.snapshotFileOffset = 0;
1898  peer.lastSnapshotIndex = 0;
1899  }
1900  }

1905 {

1907  NOTICE("Now leader for term %lu", currentTerm);
1908  state = State::LEADER;
1909  leaderId = serverId;
1910  startElectionAt = TimePoint::max();
1911  withholdVotesUntil = TimePoint::max();

1914  // lastAgreeIndex for ourselves and each follower, then append the no op.

1923  entry.set_term(currentTerm);
1924  entry.set_type(Protocol::Raft::EntryType::NOOP);

1964 void
1965 RaftConsensus::readSnapshot()
1966 {
1967  std::unique_ptr<Storage::SnapshotFile::Reader> reader;
1968  if (storageDirectory.fd != -1) {
1969  try {
1970  reader.reset(new Storage::SnapshotFile::Reader(storageDirectory));
1971  } catch (const std::runtime_error& e) { // file not found
1972  NOTICE("%s", e.what());
1973  }
1974  }
1975  if (reader) {
1976  google::protobuf::io::CodedInputStream& stream = reader->getStream();
1977 
1978  // read header protobuf from stream
1979  bool ok = true;
1980  uint32_t numBytes = 0;
1981  ok = stream.ReadLittleEndian32(&numBytes);
1982  if (!ok)
1983  PANIC("couldn't read snapshot");
1984  SnapshotMetadata::Header header;
1985  auto limit = stream.PushLimit(numBytes);
1986  ok = header.MergePartialFromCodedStream(&stream);
1987  stream.PopLimit(limit);
1988  if (!ok)
1989  PANIC("couldn't read snapshot");
1990 
1991  // load header contents
1992  if (header.last_included_index() < lastSnapshotIndex) {
1993  PANIC("Trying to load a snapshot that is more stale than one this "
1994  "server loaded earlier. The earlier snapshot covers through "
1995  "log index %lu (inclusive); this one covers through log "
1996  "index %lu (inclusive)",
1997  lastSnapshotIndex,
1998  header.last_included_index());
1999 
2000  }
2001  lastSnapshotIndex = header.last_included_index();
2002  lastSnapshotTerm = header.last_included_term();
2003  lastSnapshotBytes = reader->getSizeBytes();
2004  commitIndex = std::max(lastSnapshotIndex, commitIndex);
2005 
2006  NOTICE("Reading snapshot which covers log entries 1 through %lu "
2007  "(inclusive)", lastSnapshotIndex);
2008 
2009  // We should keep log entries if they might be needed for a quorum. So:
2010  // 1. Discard log if it is shorter than the snapshot.
2011  // 2. Discard log if its lastSnapshotIndex entry disagrees with the
2012  // lastSnapshotTerm.
2013  if (log->getLastLogIndex() < lastSnapshotIndex ||
2014  (log->getLogStartIndex() <= lastSnapshotIndex &&
2015  log->getEntry(lastSnapshotIndex).term() != lastSnapshotTerm)) {
2016  // The NOTICE message can be confusing if the log is empty, so
2017  // don't print it in that case. We still want to shift the log
2018  // start index, though.
2019  if (log->getLogStartIndex() <= log->getLastLogIndex()) {
2020  NOTICE("Discarding the entire log, since it's not known to be "
2021  "consistent with the snapshot that is being read");
2022  }
2023  // Discard the entire log, setting the log start to point to the
2024  // right place.
2025  log->truncatePrefix(lastSnapshotIndex + 1);
2026  log->truncateSuffix(lastSnapshotIndex);
2027  configurationManager->truncatePrefix(lastSnapshotIndex + 1);
2028  configurationManager->truncateSuffix(lastSnapshotIndex);
2029  }
2030 
2031  discardUnneededEntries();
2032 
2033  if (header.has_configuration_index() && header.has_configuration()) {
2034  configurationManager->setSnapshot(header.configuration_index(),
2035  header.configuration());
2036  } else {
2037  WARNING("No configuration. This is unexpected, since any snapshot "
2038  "should contain a configuration (they're the first thing "
2039  "found in any log).");
2040  }
2041 
2042  stateChanged.notify_all();
2043  }
2044  if (log->getLogStartIndex() > lastSnapshotIndex + 1) {
2045  PANIC("The newest snapshot on this server covers up through log index "
2046  "%lu (inclusive), but its log starts at index %lu. This "
2047  "should never happen and indicates a corrupt disk state. If you "
2048  "want this server to participate in your cluster, you should "
2049  "back up all of its state, delete it, and add the server back "
