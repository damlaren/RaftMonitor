77 void
78 OpaqueServerRPC::sendReply()
79 {
80  std::shared_ptr<OpaqueServer::ServerMessageSocket> socket =
81  messageSocket.lock();
82  if (socket) {
83  socket->sendMessage(messageId, std::move(response));
84  } else {
85  // During normal operation, this indicates that either the socket has
86  // been disconnected or the reply has already been sent.
87 
88  // For unit testing only, we can store replies from mock RPCs
89  // that have no sessions.
90  if (responseTarget != NULL) {
91  *responseTarget = std::move(response);
92  responseTarget = NULL;
93  }
94 
95  // Drop the reply on the floor.
96  response.reset();
97  }
98  // Prevent the server from replying again.
99  messageSocket.reset();
100 }
