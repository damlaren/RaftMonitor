31 /**
32  * This is the first part of the request header that RPC clients send, common
33  * to all versions of the protocol. RPC servers can always expect to receive
34  * this and RPC clients must always send this.
35  * This needs to be separate struct because when a server receives a request,
36  * it does not know the type of the request header, as that depends on its
37  * version.
38  */
39 struct RequestHeaderPrefix {
40  /**
41  * Convert the contents to host order from big endian (how this header
42  * should be transferred on the network).
43  */

46  * Convert the contents to big endian (how this header should be
47  * transferred on the network) from host order.
48  */
49  void toBigEndian();

51  /**
52  * This is the version of the protocol. It should always be set to 1 for
53  * now.
54  */

57 } __attribute__((packed));

106 /**
107  * The status codes returned in server responses.
108  */
109 enum class Status : uint8_t {
110 
111  /**
112  * The service processed the request and returned a valid protocol buffer
113  * with the results.
114  */
115  OK = 0,
116 
117  /**
118  * An error specific to the particular service. The format of the remainder
119  * of the message is specific to the particular service.
120  */
121  SERVICE_SPECIFIC_ERROR = 1,
122 
123  /**
124  * The server did not like the version number provided in the request
125  * header. If the client gets this, it should fall back to an older version
126  * number or crash.
127  */
128  INVALID_VERSION = 2,
129 
130  /**
131  * The server does not have the requested service.
132  */
133  INVALID_SERVICE = 3,
134 
135  /**
136  * The server did not like the RPC request. Either it specified an opCode
137  * the server didn't understand or a request protocol buffer the server
138  * couldn't accept. The client should avoid ever getting this by
139  * negotiating with the server about which version of the RPC protocol to
140  * use.
141  */
142  INVALID_REQUEST = 4,
143 
144 };
